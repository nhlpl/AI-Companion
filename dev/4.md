#![cfg_attr(not(debug_assertions), windows_subsystem = "windows")]

use std::collections::{HashMap, VecDeque, HashSet};
use std::sync::Arc;
use std::sync::atomic::{AtomicUsize, Ordering};
use tokio::sync::Mutex;
use tauri::Manager;
use serde::{Serialize, Deserialize};
use rand::Rng;
use lru::LruCache;
use std::num::NonZeroUsize;
use rayon::prelude::*;

// ------------------------------
// Constants (optimized)
// ------------------------------
const DIM: usize = 32;                // reduced from 128 (4x speed)
const MAX_MEM: usize = 25000;
const PRUNE_THRESHOLD: u16 = 655;     // 0.01 importance (u16 scale)
const EARLY_STOP_THRESHOLD: f32 = 0.3;
const CACHE_SIZE: usize = 100;
const DUPLICATE_CHECK_WINDOW: usize = 100;

// ------------------------------
// Embedding (32‑dim, hash‑based)
// ------------------------------
fn embed_text(text: &str) -> Vec<f32> {
    let mut hasher = std::collections::hash_map::DefaultHasher::new();
    std::hash::Hash::hash(&text, &mut hasher);
    let seed = std::hash::Hasher::finish(&hasher);
    let mut rng = rand::rngs::StdRng::seed_from_u64(seed);
    (0..DIM).map(|_| rng.gen_range(-1.0..1.0)).collect()
}

// ------------------------------
// Sign hash for fast similarity (popcount)
// ------------------------------
fn sign_hash(vec: &[f32]) -> u32 {
    let mut bits = 0u32;
    for (i, &v) in vec.iter().enumerate().take(32) {
        if v > 0.0 { bits |= 1 << i; }
    }
    bits
}

fn hamming_similarity(bits1: u32, bits2: u32) -> f32 {
    let diff = (bits1 ^ bits2).count_ones();
    1.0 - (diff as f32 / 32.0)
}

// ------------------------------
// Memory
// ------------------------------
#[derive(Serialize, Deserialize, Clone)]
struct Memory {
    id: usize,
    text: String,
    sign_hash: u32,           // 32‑bit sign hash for fast retrieval
    top_indices: Vec<usize>,  // top 5 absolute indices for pre‑filtering
    timestamp: f64,
    importance: u16,          // fixed‑point (0–65535)
    permanent: bool,
}

// ------------------------------
// Anti‑Memory
// ------------------------------
#[derive(Serialize, Deserialize, Clone)]
struct AntiMemory {
    incorrect: String,
    correct: String,
    successes: u32,
    failures: u32,
    created: f64,
    strength: f64,
}

// ------------------------------
// Personality (scalar 0..15)
// ------------------------------
#[derive(Serialize, Deserialize, Clone)]
struct Personality {
    value: u8,
}

// ------------------------------
// Core State
// ------------------------------
struct Core {
    memories: Vec<Memory>,
    anti_memories: Vec<AntiMemory>,
    personality: Personality,
    next_id: usize,
    cache: LruCache<u64, Vec<usize>>,   // query hash -> memory ids
}

impl Core {
    async fn load() -> Self {
        let mut core = Self {
            memories: Vec::new(),
            anti_memories: Vec::new(),
            personality: Personality { value: 7 },
            next_id: 0,
            cache: LruCache::new(NonZeroUsize::new(CACHE_SIZE).unwrap()),
        };
        if let Ok(data) = std::fs::read("data.bin") {
            if let Ok((memories, anti_memories, personality, next_id)) = bincode::deserialize(&data) {
                core.memories = memories;
                core.anti_memories = anti_memories;
                core.personality = personality;
                core.next_id = next_id;
            }
        }
        core
    }
    async fn save(&self) {
        let data = bincode::serialize(&(&self.memories, &self.anti_memories, &self.personality, self.next_id)).unwrap();
        std::fs::write("data.bin", data).unwrap();
    }

    // Add memory with duplicate detection (last 100 only)
    fn add_memory(&mut self, text: String, importance_f32: f32, permanent: bool) -> usize {
        let importance = (importance_f32 * 65535.0) as u16;
        // Duplicate check: exact text in last 100 memories
        let start = self.memories.len().saturating_sub(DUPLICATE_CHECK_WINDOW);
        for mem in &self.memories[start..] {
            if mem.text == text {
                if importance > mem.importance {
                    // update existing
                    let id = mem.id;
                    let emb = embed_text(&text);
                    let sign_hash = sign_hash(&emb);
                    let top_indices = top_indices(&emb, 5);
                    let mem = Memory { id, text, sign_hash, top_indices, timestamp: now_secs(), importance, permanent };
                    if let Some(pos) = self.memories.iter().position(|m| m.id == id) {
                        self.memories[pos] = mem;
                    }
                }
                return mem.id;
            }
        }
        // New memory
        let id = self.next_id;
        self.next_id += 1;
        let emb = embed_text(&text);
        let sign_hash = sign_hash(&emb);
        let top_indices = top_indices(&emb, 5);
        let mem = Memory { id, text, sign_hash, top_indices, timestamp: now_secs(), importance, permanent };
        self.memories.push(mem);
        // Prune low‑importance immediately
        if !permanent && importance < PRUNE_THRESHOLD {
            self.memories.retain(|m| m.permanent || m.importance >= PRUNE_THRESHOLD);
        }
        // Keep size bounded
        if self.memories.len() > MAX_MEM {
            self.memories.sort_by(|a,b| a.importance.cmp(&b.importance));
            self.memories.truncate(MAX_MEM);
        }
        id
    }

    // Retrieve top‑k memories with early termination and pre‑filtering
    fn retrieve(&self, query: &str, top_k: usize) -> Vec<String> {
        let query_emb = embed_text(query);
        let query_hash = sign_hash(&query_emb);
        let query_top = top_indices(&query_emb, 5);
        let query_hash_u64 = query_hash as u64;
        // Cache check
        if let Some(cached) = self.cache.get(&query_hash_u64) {
            return cached.iter()
                .filter_map(|&id| self.memories.iter().find(|m| m.id == id))
                .take(top_k)
                .map(|m| m.text.clone())
                .collect();
        }
        // Pre‑filter: only candidates sharing at least one top index
        let candidates: Vec<&Memory> = self.memories.iter()
            .filter(|m| m.top_indices.iter().any(|idx| query_top.contains(idx)))
            .collect();
        // Compute similarities in parallel
        let mut scored: Vec<(f32, &Memory)> = candidates.par_iter()
            .map(|&m| (hamming_similarity(query_hash, m.sign_hash), m))
            .collect();
        scored.sort_by(|a,b| b.0.partial_cmp(&a.0).unwrap());
        // Early termination: stop after we have top_k with distance < threshold
        let mut results = Vec::new();
        for (sim, mem) in scored {
            results.push(mem);
            if results.len() >= top_k && sim > (1.0 - EARLY_STOP_THRESHOLD) {
                break;
            }
        }
        // Cache results
        let ids: Vec<usize> = results.iter().map(|m| m.id).collect();
        self.cache.put(query_hash_u64, ids);
        results.into_iter().map(|m| m.text.clone()).collect()
    }

    fn add_anti_memory(&mut self, incorrect: String, correct: String) {
        self.anti_memories.push(AntiMemory {
            incorrect, correct,
            successes: 1, failures: 1,
            created: now_secs(),
            strength: 1.0,
        });
    }
    fn get_warnings(&self) -> Vec<String> {
        self.anti_memories.iter()
            .filter(|a| a.strength > 0.5)
            .map(|a| format!("Do NOT say '{}'. Instead say '{}'.", a.incorrect, a.correct))
            .collect()
    }
    fn update_personality(&mut self, delta: i8) {
        let new = (self.personality.value as i16 + delta as i16).clamp(0, 15);
        self.personality.value = new as u8;
    }
    fn export(&self) -> Vec<u8> {
        bincode::serialize(&(&self.memories, &self.anti_memories, &self.personality, self.next_id)).unwrap()
    }
    fn import(&mut self, data: &[u8]) -> Result<(), String> {
        let (memories, anti_memories, personality, next_id): (Vec<Memory>, Vec<AntiMemory>, Personality, usize) = bincode::deserialize(data).map_err(|e| e.to_string())?;
        self.memories = memories;
        self.anti_memories = anti_memories;
        self.personality = personality;
        self.next_id = next_id;
        Ok(())
    }
}

// Helper: top indices by absolute value
fn top_indices(vec: &[f32], n: usize) -> Vec<usize> {
    let mut indices: Vec<_> = (0..vec.len()).collect();
    indices.sort_by(|&a, &b| vec[b].abs().partial_cmp(&vec[a].abs()).unwrap());
    indices.into_iter().take(n).collect()
}

fn now_secs() -> f64 {
    std::time::SystemTime::now().duration_since(std::time::UNIX_EPOCH).unwrap().as_secs_f64()
}

// PersonalityProfile (unchanged)
struct PersonalityProfile;
impl PersonalityProfile {
    fn from_messages(messages: &[String]) -> f32 {
        let mut formal = 0; let mut informal = 0; let mut total_words = 0; let mut unique = std::collections::HashSet::new();
        let mut positive = 0; let mut negative = 0; let mut total_chars = 0;
        for msg in messages {
            total_chars += msg.len();
            for word in msg.split_whitespace() {
                total_words += 1;
                unique.insert(word);
                let low = word.to_lowercase();
                if low.contains("please") || low.contains("thank") { formal += 1; }
                if low.contains("hey") || low.contains("yeah") { informal += 1; }
                if low.contains("love") || low.contains("good") { positive += 1; }
                if low.contains("hate") || low.contains("bad") { negative += 1; }
            }
        }
        let formality = if formal+informal>0 { formal as f32 / (formal+informal) as f32 } else { 0.5 };
        let creativity = if total_words>0 { unique.len() as f32 / total_words as f32 } else { 0.5 };
        let empathy = if positive+negative>0 { positive as f32 / (positive+negative) as f32 } else { 0.5 };
        let verbosity = (total_chars as f32 / messages.len() as f32 / 200.0).min(1.0);
        (formality + creativity + empathy + verbosity) / 4.0
    }
}

## ✅ Optimizations Applied

| Optimization | Implementation |
|--------------|----------------|
| Embedding dimension 32 | `const DIM = 32` |
| Sign hash + popcount | `sign_hash()` + `hamming_similarity()` |
| Early termination | `if results.len() >= top_k && sim > 0.7` |
| Importance‑based immediate pruning | `if importance < PRUNE_THRESHOLD` on add |
| Parallel retrieval | `rayon::par_iter()` |
| LRU query cache | `LruCache<u64, Vec<usize>>` |
| Fixed‑point importance | `u16` (0–65535) |
| Duplicate check window | Only last 100 memories |
| Top‑index pre‑filtering | `top_indices()` and intersection check |
