// ------------------------------
// Optimized constants
// ------------------------------
const DIM: usize = 32;
const MAX_MEM: usize = 25000;
const PRUNE_THRESHOLD: u16 = 655;          // I_min = 0.01
const EARLY_STOP_THRESHOLD: f32 = 0.3;
const CACHE_SIZE: usize = 100;
const DUPLICATE_CHECK_WINDOW: usize = 100;
const ANTI_DECAY_RATE: f32 = 0.95;          // harmony: exponential decay
const ANTI_STRENGTHEN_RATE: f32 = 1.1;      // capped at 1.0
const SERENDIPITY_CHANCE: f64 = 0.05;       // 5% chance to return a random low-importance memory

// ------------------------------
// Embedding (hash‑based, 32‑dim) – now with temperature
// ------------------------------
fn embed_text(text: &str, temperature: f32) -> Vec<f32> {
    let mut hasher = std::collections::hash_map::DefaultHasher::new();
    std::hash::Hash::hash(&text, &mut hasher);
    let seed = std::hash::Hasher::finish(&hasher);
    let mut rng = rand::rngs::StdRng::seed_from_u64(seed);
    (0..DIM)
        .map(|_| {
            let base = rng.gen_range(-1.0..1.0);
            if temperature > 0.0 {
                let noise = rand::thread_rng().gen_range(-temperature..temperature);
                (base + noise).clamp(-1.0, 1.0)
            } else {
                base
            }
        })
        .collect()
}

fn sign_hash(vec: &[f32]) -> u32 {
    let mut bits = 0u32;
    for (i, &v) in vec.iter().enumerate().take(32) {
        if v > 0.0 { bits |= 1 << i; }
    }
    bits
}

fn hamming_similarity(bits1: u32, bits2: u32) -> f32 {
    let diff = (bits1 ^ bits2).count_ones();
    1.0 - (diff as f32 / 32.0)
}

struct Core {

    anti_memories: Vec<AntiMemory>,
    personality: Personality,

impl Core {
    async fn load() -> Self {
        let mut core = Self {
            memories: Vec::new(),
            anti_memories: Vec::new(),
            personality: Personality { value: 7 },
            settings: Settings::default(),

            }
        }

    }
    async fn save(&self) {
        let data = bincode::serialize(&(
            &self.memories,
            &self.anti_memories,
            &self.personality,
            self.next_id,
            &self.settings,
        ))
        .unwrap();
        std::fs::write("data.bin", data).unwrap();
    }

    fn add_memory(&mut self, text: String, importance_f32: f32, permanent: bool) -> usize {
        let importance = (importance_f32 * 65535.0) as u16;
        // Duplicate check (last 100)
        let start = self.memories.len().saturating_sub(DUPLICATE_CHECK_WINDOW);
        for mem in &self.memories[start..] {
            if mem.text == text {
                if importance > mem.importance {
                    let id = mem.id;
                    let emb = embed_text(&text, self.settings.temperature);
                    let sign_hash = sign_hash(&emb);
                    let top_indices = top_indices(&emb, 5);
                    let mem = Memory { id, text, sign_hash, top_indices, timestamp: now_secs(), importance, permanent };
                    if let Some(pos) = self.memories.iter().position(|m| m.id == id) {
                        self.memories[pos] = mem;
                    }
                }
                return mem.id;
            }
        }
        let id = self.next_id;
        self.next_id += 1;
        let emb = embed_text(&text, self.settings.temperature);
        let sign_hash = sign_hash(&emb);
        let top_indices = top_indices(&emb, 5);
        let mem = Memory { id, text, sign_hash, top_indices, timestamp: now_secs(), importance, permanent };
        self.memories.push(mem);
        // Prune low-importance if not permanent
        if !permanent && importance < PRUNE_THRESHOLD {
            self.memories.retain(|m| m.permanent || m.importance >= PRUNE_THRESHOLD);
        }
        if self.memories.len() > MAX_MEM {
            self.memories.sort_by(|a, b| a.importance.cmp(&b.importance));
            self.memories.truncate(MAX_MEM);
        }
        id
    }

    // Harmony scoring: similarity + importance + recency
    fn harmony_score(&self, mem: &Memory, sim: f32, current_time: f64) -> f32 {
        if !self.settings.harmony_mode {
            return sim;
        }
        let recency = 1.0 / (1.0 + (current_time - mem.timestamp) / 86400.0); // day half-life
        let importance = mem.importance as f32 / 65535.0;
        0.5 * sim + 0.3 * importance + 0.2 * recency
    }

    fn retrieve(&mut self, query: &str, top_k: usize) -> Vec<String> {
        let current_time = now_secs();
        let query_emb = embed_text(query, self.settings.temperature);
        let query_hash = sign_hash(&query_emb);
        let query_top = top_indices(&query_emb, 5);
        let query_hash_u64 = query_hash as u64;

        // Check cache
        if let Some(cached) = self.cache.get(&query_hash_u64) {
            return cached
                .iter()
                .filter_map(|&id| self.memories.iter().find(|m| m.id == id))
                .take(top_k)
                .map(|m| m.text.clone())
                .collect();
        }

        let candidates: Vec<&Memory> = self.memories
            .iter()
            .filter(|m| m.top_indices.iter().any(|idx| query_top.contains(idx)))
            .collect();

        let mut scored: Vec<(f32, &Memory)> = candidates
            .par_iter()
            .map(|&m| {
                let sim = hamming_similarity(query_hash, m.sign_hash);
                let score = self.harmony_score(m, sim, current_time);
                (score, m)
            })
            .collect();

        scored.sort_by(|a, b| b.0.partial_cmp(&a.0).unwrap());

        // Serendipity: occasionally inject a random low-importance memory
        if self.settings.harmony_mode && rand::thread_rng().gen_bool(SERENDIPITY_CHANCE) {
            let low_imp: Vec<_> = self.memories.iter().filter(|m| m.importance < PRUNE_THRESHOLD && !m.permanent).collect();
            if let Some(&random_mem) = low_imp.get(rand::thread_rng().gen_range(0..low_imp.len())) {
                scored.insert(1, (0.8, random_mem)); // inject near top
            }
        }

        let mut results = Vec::new();
        for (score, mem) in scored {
            results.push(mem);
            if results.len() >= top_k && score > (1.0 - EARLY_STOP_THRESHOLD) {
                break;
            }
        }

        let ids: Vec<usize> = results.iter().map(|m| m.id).collect();
        self.cache.put(query_hash_u64, ids);
        results.into_iter().map(|m| m.text.clone()).collect()
    }

    fn add_anti_memory(&mut self, incorrect: String, correct: String) {
        self.anti_memories.push(AntiMemory {
            incorrect,
            correct,
            successes: 1,
            failures: 1,
            created: now_secs(),
            strength: 1.0,
        });
    }

    // Harmony: decay or strengthen anti-memory based on success
    fn reinforce_anti_memory(&mut self, incorrect_phrase: &str, success: bool) {
        if let Some(am) = self.anti_memories.iter_mut().find(|a| a.incorrect == incorrect_phrase) {
            if success {
                am.successes += 1;
                am.strength *= ANTI_DECAY_RATE;
            } else {
                am.failures += 1;
                am.strength = (am.strength * ANTI_STRENGTHEN_RATE).min(1.0);
            }
        }
    }

    fn get_warnings(&self) -> Vec<String> {
        self.anti_memories
            .iter()
            .filter(|a| a.strength > 0.5)
            .map(|a| format!("Do NOT say '{}'. Instead say '{}'.", a.incorrect, a.correct))
            .collect()
    }

    fn update_personality(&mut self, delta: i8) {
        let new = (self.personality.value as i16 + delta as i16).clamp(0, 15);
        self.personality.value = new as u8;
    }

    fn export(&self) -> Vec<u8> {
        bincode::serialize(&(
            &self.memories,
            &self.anti_memories,
            &self.personality,
            self.next_id,
            &self.settings,
        ))
        .unwrap()
    }

    fn import(&mut self, data: &[u8]) -> Result<(), String> {
        let (memories, anti_memories, personality, next_id, settings): (
            Vec<Memory>,
            Vec<AntiMemory>,
            Personality,
            usize,
            Settings,
        ) = bincode::deserialize(data).map_err(|e| e.to_string())?;
        self.memories = memories;
        self.anti_memories = anti_memories;
        self.personality = personality;
        self.next_id = next_id;
        self.settings = settings;
        Ok(())
    }

    fn update_settings(&mut self, temperature: Option<f32>, harmony_mode: Option<bool>) {
        if let Some(t) = temperature {
            self.settings.temperature = t.clamp(0.0, 1.0);
        }
        if let Some(h) = harmony_mode {
            self.settings.harmony_mode = h;
        }
    }
}

fn top_indices(vec: &[f32], n: usize) -> Vec<usize> {
    let mut indices: Vec<_> = (0..vec.len()).collect();
    indices.sort_by(|&a, &b| vec[b].abs().partial_cmp(&vec[a].abs()).unwrap());
    indices.into_iter().take(n).collect()
}

        .unwrap()
        .as_secs_f64()
}

struct PersonalityProfile;
impl PersonalityProfile {
    fn from_messages(messages: &[String]) -> f32 {
        let mut formal = 0; let mut informal = 0; let mut total_words = 0; let mut unique = HashSet::new();
        let mut positive = 0; let mut negative = 0; let mut total_chars = 0;
        for msg in messages {
            total_chars += msg.len();
            for word in msg.split_whitespace() {
                total_words += 1;
                unique.insert(word);
                let low = word.to_lowercase();
                if low.contains("please") || low.contains("thank") { formal += 1; }
                if low.contains("hey") || low.contains("yeah") { informal += 1; }
                if low.contains("love") || low.contains("good") { positive += 1; }
                if low.contains("hate") || low.contains("bad") { negative += 1; }
            }
        }
        let formality = if formal+informal>0 { formal as f32 / (formal+informal) as f32 } else { 0.5 };
        let creativity = if total_words>0 { unique.len() as f32 / total_words as f32 } else { 0.5 };
        let empathy = if positive+negative>0 { positive as f32 / (positive+negative) as f32 } else { 0.5 };
        let verbosity = (total_chars as f32 / messages.len() as f32 / 200.0).min(1.0);
        (formality + creativity + empathy + verbosity) / 4.0
    }
}

        async setActiveTab(tab) { this.activeTab = tab; if (tab === 'chat') await this.loadRecentMessages(); if (tab === 'settings') await this.loadSettings(); },
        async sendMessage() {
            if (isSending) return;
            const msg = this.inputMessage.trim();
            if (!msg) return;
            isSending = true; this.isLoading = true;
            this.chatMessages.push({ role: 'user', content: msg });
            this.inputMessage = '';
            document.getElementById('chatLog').scrollTop = document.getElementById('chatLog').scrollHeight;
            try {
                const response = await window.__TAURI__.invoke('send_message', { message: msg });
                this.chatMessages.push({ role: 'ai', content: response });
                document.getElementById('chatLog').scrollTop = document.getElementById('chatLog').scrollHeight;
                await this.loadRecentMessages();
            } catch (err) {
                this.chatMessages.push({ role: 'ai', content: `Error: ${err}` });
            } finally { this.isLoading = false; isSending = false; }
        },
        async loadRecentMessages() {
            const memories = await window.__TAURI__.invoke('get_recent_memories', { k: 30 });
            const msgs = [];
            for (const mem of memories) {
                if (mem.startsWith('User:')) {
                    msgs.push({ role: 'user', content: mem.split('\n')[0].replace('User:', '').trim() });
                    const ai = mem.split('\nAI:')[1];
                    if (ai) msgs.push({ role: 'ai', content: ai.trim() });
                }
            }
            this.chatMessages = msgs.slice(-30);
        },

            await window.__TAURI__.invoke('update_personality', { delta });
            const p = await window.__TAURI__.invoke('get_personality');
            this.personalityHex = p.hex;
        },
        async addAntiMemory() {
            if (!this.antiIncorrect || !this.antiCorrect) return;

            const file = document.getElementById('chatFile').files[0];
            if (!file) return;
            const reader = new FileReader();
            reader.onload = async (e) => {
                this.profileResult = await window.__TAURI__.invoke('analyze_chat_text', { text: e.target.result });
            };
            reader.readAsText(file);
        },
        async applyProfileHex() {
            if (!this.profileResult) return;
            await window.__TAURI__.invoke('apply_personality', { hex: this.profileResult });
            const p = await window.__TAURI__.invoke('get_personality');
            this.personalityHex = p.hex;
            alert('Personality applied');

            const data = await window.__TAURI__.invoke('export_profile');
            const blob = new Blob([new Uint8Array(data)], { type: 'application/octet-stream' });

            const a = document.createElement('a');
            a.href = url; a.download = `ai_backup_${new Date().toISOString()}.aibak`;
            a.click(); URL.revokeObjectURL(url);
        },

            const file = document.getElementById('restoreFile').files[0];
            if (!file) return;
            const reader = new FileReader();
            reader.onload = async (e) => {
                const bytes = new Uint8Array(e.target.result);
                const result = await window.__TAURI__.invoke('import_profile', { data: Array.from(bytes) });
                alert(result); await this.loadRecentMessages(); await this.loadSettings();
            };
            reader.readAsArrayBuffer(file);
        },
        async loadSettings() {
            const s = await window.__TAURI__.invoke('get_settings');
            this.settings = s;
        },
        async updateSettingsDebounced() {
            clearTimeout(this._settingsTimeout);
            this._settingsTimeout = setTimeout(async () => {
                await window.__TAURI__.invoke('update_settings', {
                    temperature: this.settings.temperature,
                    harmony_mode: this.settings.harmony_mode,
                });
            }, 300);
        },
        async loadInitial() {
            const p = await window.__TAURI__.invoke('get_personality');
            this.personalityHex = p.hex;
            await this.loadSettings();
            await this.loadRecentMessages();
        }
    },
    mounted() { this.loadInitial(); }
});

## ✅ Summary of Harmony Enhancements

| Feature | Implementation | Benefit |
|:---|:---|:---|
| **Temperature** | `embed_text` now accepts a temperature parameter (0–0.5). Controlled via slider. | Introduces controlled creativity without identity loss. |
| **Harmony Scoring** | Retrieval uses 50% similarity + 30% importance + 20% recency. | More human-like memory, balancing relevance and freshness. |
| **Serendipity** | 5% chance to inject a random low-importance memory. | Surprise and delight; prevents rigid conversation. |
| **Anti-Memory Decay** | `reinforce_anti_memory` command decays strength on success. | Learned corrections fade when no longer needed. |
| **Settings Persistence** | Temperature and harmony mode saved in `data.bin`. | User preferences survive restarts. |

The app remains **small**—no new heavy dependencies, all logic fits in ~400 lines of Rust and a compact HTML/JS frontend. Harmony is achieved through algorithmic balance, not code bloat.
