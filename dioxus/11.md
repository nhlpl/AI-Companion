#[derive(Clone)]
pub struct StdpBatchEntry {
    pub memory_id: usize,
    pub retrieval_time: f64,
}

pub struct StdpUpdater {
    pending: VecDeque<StdpBatchEntry>,
    pot_lookup: Vec<f32>,
    dep_lookup: Vec<f32>,
}

impl StdpUpdater {
    pub fn new() -> Self {
        let steps = 1000;
        let max_dt = STDP_MAX_DT;
        let mut pot_lookup = Vec::with_capacity(steps);
        let mut dep_lookup = Vec::with_capacity(steps);
        for i in 0..steps {
            let dt = (i as f32 / steps as f32) * max_dt;
            pot_lookup.push(STDP_A_PLUS * (-dt / STDP_TAU_PLUS).exp());
            dep_lookup.push(-STDP_A_MINUS * (dt / STDP_TAU_MINUS).exp());
        }
        Self { pending: VecDeque::new(), pot_lookup, dep_lookup }
    }

    pub fn record(&mut self, memory_id: usize, retrieval_time: f64) {
        self.pending.push_back(StdpBatchEntry { memory_id, retrieval_time });
        if self.pending.len() > 200 { self.pending.pop_front(); }
    }

    pub fn apply_pending<F>(&mut self, action_time: f64, mut update_fn: F)
    where F: FnMut(usize, f32)
    {
        let steps = self.pot_lookup.len();
        let max_dt = STDP_MAX_DT;
        while let Some(entry) = self.pending.pop_front() {
            let dt = action_time - entry.retrieval_time;
            if dt.abs() > max_dt as f64 { continue; }
            let idx = ((dt.abs() as f32 / max_dt) * (steps - 1) as f32).round() as usize;
            let delta = if dt > 0.0 { self.pot_lookup[idx] } else { self.dep_lookup[idx] };
            update_fn(entry.memory_id, delta);
        }
    }
}

pub fn analyze_sentiment(msgs: &[&str], locale: Locale) -> f32 {
    let (pos_w, neg_w): (&[&str], &[&str]) = match locale {
        Locale::En => (
            &["good","great","happy","love","excited","wonderful","nice"],
            &["bad","sad","angry","hate","terrible","upset","awful"],
        ),
        Locale::Zh => (
            &["好","棒","开心","喜欢","爱","优秀","不错","高兴","快乐"],
            &["坏","差","难过","讨厌","恨","糟糕","烦","生气","伤心"],
        ),
    };
    let mut pos = 0;
    let mut neg = 0;
    for msg in msgs {
        for word in msg.split_whitespace() {
            let low = word.to_lowercase();
            if pos_w.iter().any(|&pw| low.contains(pw)) { pos += 1; }
            if neg_w.iter().any(|&nw| low.contains(nw)) { neg += 1; }
        }
        // Chinese: also check substrings (no spaces)
        if matches!(locale, Locale::Zh) {
            for &pw in pos_w { if msg.contains(pw) { pos += 1; } }
            for &nw in neg_w { if msg.contains(nw) { neg += 1; } }
        }
    }
    if pos + neg == 0 { 0.5 } else { pos as f32 / (pos + neg) as f32 }
}

    for r in responses {
        for w in r.split_whitespace() {
            let wc = w.to_lowercase().chars().filter(|c| c.is_alphabetic()).collect::<String>();
            if wc.len() > 4 && !stopwords.contains(wc.as_str()) {

                    if !stopwords.contains(s.as_str()) {

                    }
                }
            }
        }
