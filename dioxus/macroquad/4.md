We've heard the feedback loud and clear: **sliders are boring. Interactive, playful avatars are memorable.** The living avatar should be the star of the show—something users want to click, drag, and watch react. This turns a passive screensaver into an engaging companion.

## 🎮 New Interactive Avatar Behaviors

| Interaction | Visual Response | Psychological Effect |
| :--- | :--- | :--- |
| **Hover** | Subtle glow, slight scale increase (1.05x) | "It sees me." Creates a sense of presence and attention. |
| **Click (Tap)** | Happy pulse, sparkle particles, temporary hue shift toward gold | "I made it happy!" Positive reinforcement, playful engagement. |
| **Double-Click** | Avatar spins joyfully, emits a burst of colorful particles | "We're playing!" Delight, surprise. |
| **Drag** | Avatar follows mouse with a gentle elastic delay; rotation aligns to drag direction | "I can move it!" Sense of control and physicality. |
| **Idle Petting** (slow drag) | Soft, rhythmic pulse; warm glow | Comfort, bonding. |

### `avatar/src/main.rs` (Complete Interactive Version)

```rust
use macroquad::prelude::*;
use serde::{Deserialize, Serialize};
use std::collections::VecDeque;
use std::f32::consts::PI;
use std::sync::Arc;
use tokio::net::TcpListener;
use tokio::sync::watch;

// ---------- State from main app ----------
#[derive(Debug, Deserialize, Serialize, Clone)]
struct AvatarState {
    emotion: String,
    trust: f32,
    mode: String,
    speaking: bool,
    message_count: u32,
    memory_glow: bool,
    listening: bool,
    warning: bool,
}

impl Default for AvatarState {
    fn default() -> Self {
        Self {
            emotion: String::new(),
            trust: 0.5,
            mode: "Companion".to_string(),
            speaking: false,
            message_count: 0,
            memory_glow: false,
            listening: false,
            warning: false,
        }
    }
}

// ---------- Interactive state ----------
struct InteractiveState {
    // Position offset (for drag)
    offset_x: f32,
    offset_y: f32,
    target_offset_x: f32,
    target_offset_y: f32,

    // Drag tracking
    is_dragging: bool,
    drag_start_mouse: Vec2,
    drag_start_offset: Vec2,

    // Hover
    is_hovering: bool,
    hover_strength: f32,

    // Click effects
    click_pulse: f32,        // 0..1, decays
    click_sparkles: Vec<Sparkle>,
    double_click_triggered: bool,

    // Petting (slow drag)
    petting_active: bool,
    petting_strength: f32,

    // Elastic rotation offset
    rotation_offset: f32,
    target_rotation_offset: f32,
}

struct Sparkle {
    pos: Vec2,
    vel: Vec2,
    life: f32,
    color: Color,
}

impl InteractiveState {
    fn new() -> Self {
        Self {
            offset_x: 0.0,
            offset_y: 0.0,
            target_offset_x: 0.0,
            target_offset_y: 0.0,
            is_dragging: false,
            drag_start_mouse: Vec2::ZERO,
            drag_start_offset: Vec2::ZERO,
            is_hovering: false,
            hover_strength: 0.0,
            click_pulse: 0.0,
            click_sparkles: Vec::new(),
            double_click_triggered: false,
            petting_active: false,
            petting_strength: 0.0,
            rotation_offset: 0.0,
            target_rotation_offset: 0.0,
        }
    }

    fn update(&mut self, mouse_pos: Vec2, mouse_down: bool, mouse_pressed: bool, avatar_center: Vec2, avatar_radius: f32) {
        let dt = 0.016;
        let dist = mouse_pos.distance(avatar_center);

        // Hover detection
        let was_hovering = self.is_hovering;
        self.is_hovering = dist < avatar_radius * 1.2;
        if self.is_hovering && !was_hovering {
            // Just started hovering
        }
        self.hover_strength += (if self.is_hovering { 1.0 } else { 0.0 } - self.hover_strength) * 0.1;

        // Drag handling
        if mouse_pressed && self.is_hovering && !self.is_dragging {
            self.is_dragging = true;
            self.drag_start_mouse = mouse_pos;
            self.drag_start_offset = Vec2::new(self.target_offset_x, self.target_offset_y);
        }
        if self.is_dragging {
            if mouse_down {
                let delta = mouse_pos - self.drag_start_mouse;
                self.target_offset_x = self.drag_start_offset.x + delta.x;
                self.target_offset_y = self.drag_start_offset.y + delta.y;

                // Detect petting (slow drag speed)
                let speed = delta.length() / dt;
                self.petting_active = speed > 5.0 && speed < 100.0;
                self.petting_strength += (if self.petting_active { 1.0 } else { 0.0 } - self.petting_strength) * 0.05;
            } else {
                self.is_dragging = false;
                self.petting_active = false;
                // Spring back toward center
                self.target_offset_x = 0.0;
                self.target_offset_y = 0.0;
            }
        }

        // Elastic movement
        self.offset_x += (self.target_offset_x - self.offset_x) * 0.1;
        self.offset_y += (self.target_offset_y - self.offset_y) * 0.1;

        // Rotation offset from drag direction
        if self.is_dragging && !self.petting_active {
            let drag_dir = (mouse_pos - avatar_center).normalize_or_zero();
            self.target_rotation_offset = drag_dir.x * 0.5;
        } else {
            self.target_rotation_offset = 0.0;
        }
        self.rotation_offset += (self.target_rotation_offset - self.rotation_offset) * 0.05;

        // Click handling (outside drag start)
        if mouse_pressed && self.is_hovering && !self.is_dragging {
            self.click_pulse = 1.0;
            // Spawn sparkles
            for _ in 0..8 {
                let angle = rand::gen_range(0.0, PI * 2.0);
                let speed = rand::gen_range(50.0, 150.0);
                self.click_sparkles.push(Sparkle {
                    pos: avatar_center,
                    vel: Vec2::new(angle.cos() * speed, angle.sin() * speed),
                    life: 1.0,
                    color: Color::from_hsl(rand::gen_range(0.0, 0.2), 0.8, 0.7),
                });
            }
        }

        // Double-click detection (simplified: two clicks within 0.3s)
        // We'll implement a simple timer in the main loop.

        // Decay effects
        self.click_pulse *= 0.9;
        self.hover_strength *= 0.95;
        self.petting_strength *= 0.98;

        // Update sparkles
        for s in &mut self.click_sparkles {
            s.pos += s.vel * dt;
            s.vel *= 0.95;
            s.life -= dt * 1.5;
        }
        self.click_sparkles.retain(|s| s.life > 0.0);
    }
}

// ---------- Visual parameters ----------
struct VisualParams {
    depth: f32,
    hue: f32,
    saturation: f32,
    value: f32,
    spike_factor: f32,
    rotation_speed: f32,
    pulse_amplitude: f32,
    glow_intensity: f32,
    listening_expansion: f32,
    mode: String,
    speaking: bool,
}

impl Default for VisualParams {
    fn default() -> Self {
        Self {
            depth: 2.0,
            hue: 0.3,
            saturation: 0.5,
            value: 0.8,
            spike_factor: 0.0,
            rotation_speed: 0.4,
            pulse_amplitude: 0.0,
            glow_intensity: 0.0,
            listening_expansion: 1.0,
            mode: "Companion".to_string(),
            speaking: false,
        }
    }
}

fn emotion_params(emotion: &str) -> (f32, f32) {
    match emotion {
        "frustration" => (0.0, 0.9),
        "sadness" => (0.6, 0.0),
        "excitement" => (0.12, 0.0),
        "anxiety" => (0.75, 0.5),
        "confusion" => (0.0, 0.2),
        _ => (0.3, 0.0),
    }
}

fn draw_branch(
    x: f32, y: f32, length: f32, angle: f32, depth: u32, max_depth: u32,
    params: &VisualParams, time: f32, base_hue: f32, interactive: &InteractiveState,
) {
    if depth == 0 { return; }

    let end_x = x + length * angle.cos();
    let end_y = y + length * angle.sin();

    let depth_factor = depth as f32 / max_depth as f32;
    let pulse = if params.speaking {
        params.pulse_amplitude * (time * 8.0).sin()
    } else {
        0.0
    };
    let petting_pulse = interactive.petting_strength * 0.1 * (time * 4.0).sin();
    let lightness = (params.value + pulse * 0.1 + petting_pulse).clamp(0.3, 1.0);
    let saturation = (params.saturation + params.glow_intensity * 0.2 + interactive.hover_strength * 0.1).clamp(0.1, 1.0);
    let color = Color::from_hsl(params.hue, saturation, lightness);

    let thickness = (2.0 * (1.0 - depth_factor) + 1.0) * (1.0 + params.spike_factor * 0.5) * (1.0 + interactive.hover_strength * 0.2);
    draw_line(x, y, end_x, end_y, thickness, color);

    let new_length = length * 0.7;
    let angle_variance = match params.mode.as_str() {
        "Mentor" => PI / 5.0,
        "Trickster" => PI / 3.5,
        _ => PI / 4.5,
    };
    let spike_angle = params.spike_factor * PI / 6.0;
    let left_angle = angle - angle_variance - spike_angle;
    let right_angle = angle + angle_variance + spike_angle;

    draw_branch(end_x, end_y, new_length, left_angle, depth - 1, max_depth, params, time, base_hue, interactive);
    draw_branch(end_x, end_y, new_length, right_angle, depth - 1, max_depth, params, time, base_hue, interactive);

    if depth > 2 && max_depth > 4 {
        draw_branch(end_x, end_y, new_length * 0.6, angle, depth - 1, max_depth, params, time, base_hue, interactive);
    }
}

// ---------- Main ----------
#[macroquad::main("AI Companion Avatar")]
async fn main() {
    let base_hue: f32 = std::env::args()
        .collect::<Vec<_>>()
        .iter()
        .position(|a| a == "--hue")
        .and_then(|i| std::env::args().nth(i + 1))
        .and_then(|s| s.parse().ok())
        .unwrap_or(0.3);

    let listener = TcpListener::bind("127.0.0.1:9001").await.unwrap();
    println!("Interactive Avatar listening on 127.0.0.1:9001");

    let (tx, mut rx) = watch::channel(AvatarState::default());
    let tx = Arc::new(tx);

    tokio::spawn(async move {
        loop {
            if let Ok((mut stream, _)) = listener.accept().await {
                let tx = tx.clone();
                tokio::spawn(async move {
                    let mut buf = vec![0u8; 1024];
                    loop {
                        match stream.try_read(&mut buf) {
                            Ok(n) if n > 0 => {
                                if let Ok(state) = serde_json::from_slice::<AvatarState>(&buf[..n]) {
                                    let _ = tx.send(state);
                                }
                            }
                            _ => break,
                        }
                    }
                });
            }
        }
    });

    let mut visual = VisualParams::default();
    let mut interactive = InteractiveState::new();
    let mut time = 0.0;
    let mut rotation = 0.0;
    let mut memory_glow_decay = 0.0;
    let mut warning_flash = 0.0;

    let mut target_depth = 2.0;
    let mut target_hue = base_hue;
    let mut target_saturation = 0.5;
    let mut target_spike = 0.0;
    let mut target_speed = 0.4;
    let mut target_pulse = 0.0;

    // Double-click detection
    let mut last_click_time = 0.0;
    let double_click_window = 0.3;

    loop {
        time = get_time() as f32;
        let mouse_pos = mouse_position();
        let mouse_pos = Vec2::new(mouse_pos.0, mouse_pos.1);
        let mouse_down = is_mouse_button_down(MouseButton::Left);
        let mouse_pressed = is_mouse_button_pressed(MouseButton::Left);

        let avatar_center = Vec2::new(screen_width() / 2.0 + interactive.offset_x, screen_height() / 2.0 + 30.0 + interactive.offset_y);
        let avatar_radius = 120.0;

        // Update interactive state
        interactive.update(mouse_pos, mouse_down, mouse_pressed, avatar_center, avatar_radius);

        // Handle click for double-click detection
        if mouse_pressed && interactive.is_hovering {
            let now = time;
            if now - last_click_time < double_click_window {
                // Double click!
                interactive.double_click_triggered = true;
                // Big spin and sparkles
                interactive.target_rotation_offset += PI * 2.0;
                for _ in 0..20 {
                    let angle = rand::gen_range(0.0, PI * 2.0);
                    let speed = rand::gen_range(80.0, 200.0);
                    interactive.click_sparkles.push(Sparkle {
                        pos: avatar_center,
                        vel: Vec2::new(angle.cos() * speed, angle.sin() * speed),
                        life: 1.2,
                        color: Color::from_hsl(rand::gen_range(0.0, 1.0), 0.9, 0.8),
                    });
                }
            }
            last_click_time = now;
        }

        // Get latest state from main app
        if let Ok(state) = rx.borrow_and_update().map(|s| s.clone()) {
            let (emotion_shift, emotion_spike) = emotion_params(&state.emotion);
            target_hue = (base_hue + emotion_shift * 0.3).clamp(0.0, 1.0);
            target_saturation = 0.4 + state.trust * 0.6;
            target_depth = 2.0 + (state.message_count as f32 / 20.0).min(5.0);
            target_speed = match state.mode.as_str() {
                "Mentor" => 0.2,
                "Trickster" => 0.8,
                _ => 0.4,
            };
            if state.speaking {
                target_speed *= 1.8;
                target_pulse = 0.15;
            } else {
                target_pulse = 0.0;
            }
            target_spike = emotion_spike;
            visual.mode = state.mode.clone();
            visual.speaking = state.speaking;

            if state.memory_glow { memory_glow_decay = 1.0; }
            if state.warning { warning_flash = 1.0; }
            visual.listening_expansion = if state.listening { 1.05 } else { 1.0 };
        }

        // Smooth interpolation
        visual.depth += (target_depth - visual.depth) * 0.05;
        visual.hue += (target_hue - visual.hue) * 0.05;
        visual.saturation += (target_saturation - visual.saturation) * 0.03;
        visual.spike_factor += (target_spike - visual.spike_factor) * 0.08;
        visual.rotation_speed += (target_speed - visual.rotation_speed) * 0.05;
        visual.pulse_amplitude += (target_pulse - visual.pulse_amplitude) * 0.1;

        memory_glow_decay *= 0.95;
        warning_flash *= 0.9;
        visual.glow_intensity = memory_glow_decay + interactive.click_pulse * 0.3 + interactive.petting_strength * 0.2;

        // Rotation: base + drag influence + double-click spin
        rotation += (visual.rotation_speed + interactive.rotation_offset.abs() * 0.5) * 0.016;
        if interactive.double_click_triggered {
            rotation += 0.5;
            interactive.double_click_triggered = false;
        }

        // Render
        clear_background(Color::new(0.05, 0.05, 0.08, 1.0));

        let scale = visual.listening_expansion * (1.0 + interactive.hover_strength * 0.03) * (1.0 + interactive.click_pulse * 0.05);

        if warning_flash > 0.01 {
            draw_rectangle(0.0, 0.0, screen_width(), screen_height(), Color::new(1.0, 0.0, 0.0, warning_flash * 0.3));
        }

        let trunk_length = 100.0 * scale;
        draw_branch(
            avatar_center.x, avatar_center.y, trunk_length,
            -PI / 2.0 + rotation * 0.1 + interactive.rotation_offset,
            visual.depth.round() as u32, visual.depth.round() as u32,
            &visual, time, base_hue, &interactive,
        );

        // Draw sparkles
        for s in &interactive.click_sparkles {
            draw_circle(s.pos.x, s.pos.y, 3.0 * s.life, s.color);
        }

        // Central glow (heart)
        let heart_color = Color::from_hsl(visual.hue, visual.saturation, 0.9);
        draw_circle(avatar_center.x, avatar_center.y, 6.0 + visual.glow_intensity * 10.0, heart_color);

        // Hover indicator (subtle ring)
        if interactive.is_hovering && !interactive.is_dragging {
            draw_circle_lines(avatar_center.x, avatar_center.y, avatar_radius * 1.1, 1.5, Color::new(1.0, 1.0, 1.0, interactive.hover_strength * 0.5));
        }

        next_frame().await;
    }
}
```
