### 1. `src/domain/settings_engine.rs` – Add `AvatarColor`

#[derive(Serialize, Deserialize, Clone, PartialEq)]
pub enum AvatarColor {
    Rose,
    Ocean,
    Forest,
    Sunset,
    Midnight,
}

impl Default for AvatarColor {
    fn default() -> Self { Self::Forest }
}

impl AvatarColor {
    pub fn to_hue(&self) -> f32 {
        match self {
            Self::Rose => 0.95,
            Self::Ocean => 0.58,
            Self::Forest => 0.30,
            Self::Sunset => 0.08,
            Self::Midnight => 0.75,
        }
    }
    
    pub fn name(&self) -> &'static str {
        match self {
            Self::Rose => "Rose",
            Self::Ocean => "Ocean",
            Self::Forest => "Forest",
            Self::Sunset => "Sunset",
            Self::Midnight => "Midnight",
        }
    }
}
```

        spawn_avatar_process(&avatar_color);
        
        // ...
    }
    
    fn spawn_avatar_process(color: &AvatarColor) {
        let hue = color.to_hue();
        let result = Command::new("./avatar/ai_companion_avatar")
            .arg("--hue")
            .arg(hue.to_string())
            .spawn();
        match result {
            Ok(child) => log::info!("Avatar spawned with hue: {}", hue),
            Err(e) => log::error!("Failed to spawn avatar: {}", e),
        }
    }
}
```

fn parse_hue() -> f32 {

    if let Some(idx) = args.iter().position(|a| a == "--hue") {

### 5. Onboarding Screen (New Component) – `src/ui/onboarding.rs`

```rust
use dioxus::prelude::*;
use crate::domain::AvatarColor;
use crate::state::AppState;

#[component]
pub fn Onboarding() -> Element {
    let mut state = use_context::<AppState>();
    let mut selected = use_signal(|| AvatarColor::Forest);
    
    let colors = vec![
        AvatarColor::Rose,
        AvatarColor::Ocean,
        AvatarColor::Forest,
        AvatarColor::Sunset,
        AvatarColor::Midnight,
    ];
    
    rsx! {
        div { class: "flex flex-col items-center justify-center h-screen bg-gray-900 text-white",
            h1 { class: "text-3xl mb-8", "Choose your companion's color" }
            div { class: "flex gap-6 mb-12",
                for color in colors {
                    button {
                        class: format!("w-16 h-16 rounded-full {} border-4 {}",
                            color_class(&color),
                            if *selected.read() == color { "border-white" } else { "border-transparent" }
                        ),
                        onclick: move |_| selected.set(color.clone()),
                    }
                }
            }
            button {
                class: "px-8 py-3 bg-blue-600 rounded-full text-xl font-bold hover:bg-blue-700",
                onclick: move |_| {
                    state.set_avatar_color(selected.read().clone());
                    state.mark_onboarding_complete();
                },
                "Continue"
            }
        }
    }
}

fn color_class(color: &AvatarColor) -> &'static str {
    match color {
        AvatarColor::Rose => "bg-gradient-to-br from-pink-400 to-red-500",
        AvatarColor::Ocean => "bg-gradient-to-br from-cyan-400 to-blue-500",
        AvatarColor::Forest => "bg-gradient-to-br from-green-400 to-emerald-600",
        AvatarColor::Sunset => "bg-gradient-to-br from-yellow-400 to-orange-500",
        AvatarColor::Midnight => "bg-gradient-to-br from-purple-400 to-indigo-600",
    }
}
```
