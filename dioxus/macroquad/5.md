```rust
// ---------- Interactive State (preserved across AI updates) ----------
struct InteractiveState {
    offset_x: f32,
    offset_y: f32,
    target_offset_x: f32,
    target_offset_y: f32,
    is_dragging: bool,
    drag_start_mouse: Vec2,
    drag_start_offset: Vec2,
    is_hovering: bool,
    hover_strength: f32,
    click_pulse: f32,
    click_sparkles: Vec<Sparkle>,
    double_click_triggered: bool,
    petting_active: bool,
    petting_strength: f32,
    rotation_offset: f32,
    target_rotation_offset: f32,
    // Memory glow queue
    memory_glow_queue: f32,
}

struct Sparkle {
    pos: Vec2,
    vel: Vec2,
    life: f32,
    color: Color,
}

impl InteractiveState {
    fn new() -> Self {
        Self {
            offset_x: 0.0, offset_y: 0.0, target_offset_x: 0.0, target_offset_y: 0.0,
            is_dragging: false, drag_start_mouse: Vec2::ZERO, drag_start_offset: Vec2::ZERO,
            is_hovering: false, hover_strength: 0.0, click_pulse: 0.0, click_sparkles: Vec::new(),
            double_click_triggered: false, petting_active: false, petting_strength: 0.0,
            rotation_offset: 0.0, target_rotation_offset: 0.0, memory_glow_queue: 0.0,
        }
    }

    fn update(&mut self, mouse_pos: Vec2, mouse_down: bool, mouse_pressed: bool, avatar_center: Vec2, avatar_radius: f32) {
        let dt = 0.016;
        let dist = mouse_pos.distance(avatar_center);

        // Hover detection
        self.is_hovering = dist < avatar_radius * 1.2;
        self.hover_strength += (if self.is_hovering { 1.0 } else { 0.0 } - self.hover_strength) * 0.1;

        // Drag handling
        if mouse_pressed && self.is_hovering && !self.is_dragging {
            self.is_dragging = true;
            self.drag_start_mouse = mouse_pos;
            self.drag_start_offset = Vec2::new(self.target_offset_x, self.target_offset_y);
        }
        if self.is_dragging {
            if mouse_down {
                let delta = mouse_pos - self.drag_start_mouse;
                self.target_offset_x = self.drag_start_offset.x + delta.x;
                self.target_offset_y = self.drag_start_offset.y + delta.y;
                let speed = delta.length() / dt;
                self.petting_active = speed > 5.0 && speed < 100.0;
                self.petting_strength += (if self.petting_active { 1.0 } else { 0.0 } - self.petting_strength) * 0.05;
            } else {
                self.is_dragging = false;
                self.petting_active = false;
                self.target_offset_x = 0.0;
                self.target_offset_y = 0.0;
            }
        }

        // Elastic movement
        self.offset_x += (self.target_offset_x - self.offset_x) * 0.1;
        self.offset_y += (self.target_offset_y - self.offset_y) * 0.1;

        // Rotation offset from drag
        if self.is_dragging && !self.petting_active {
            let drag_dir = (mouse_pos - avatar_center).normalize_or_zero();
            self.target_rotation_offset = drag_dir.x * 0.5;
        } else {
            self.target_rotation_offset = 0.0;
        }
        self.rotation_offset += (self.target_rotation_offset - self.rotation_offset) * 0.05;

        // Click handling
        if mouse_pressed && self.is_hovering && !self.is_dragging {
            self.click_pulse = 1.0;
            for _ in 0..8 {
                let angle = rand::gen_range(0.0, PI * 2.0);
                let speed = rand::gen_range(50.0, 150.0);
                self.click_sparkles.push(Sparkle {
                    pos: avatar_center,
                    vel: Vec2::new(angle.cos() * speed, angle.sin() * speed),
                    life: 1.0,
                    color: Color::from_hsl(rand::gen_range(0.0, 0.2), 0.8, 0.7),
                });
            }
        }

        // Decay effects
        self.click_pulse *= 0.9;
        self.petting_strength *= 0.98;
        self.memory_glow_queue *= 0.95;

        // Update sparkles
        for s in &mut self.click_sparkles {
            s.pos += s.vel * dt;
            s.vel *= 0.95;
            s.life -= dt * 1.5;
        }
        self.click_sparkles.retain(|s| s.life > 0.0);
    }

    fn queue_memory_glow(&mut self) {
        self.memory_glow_queue = 1.0;
    }
}

// ---------- Visual Params ----------
struct VisualParams {
    depth: f32, hue: f32, saturation: f32, value: f32, spike_factor: f32,
    rotation_speed: f32, pulse_amplitude: f32, glow_intensity: f32,
    listening_expansion: f32, mode: String, speaking: bool,
}
impl Default for VisualParams {
    fn default() -> Self {
        Self { depth:2.0, hue:0.3, saturation:0.5, value:0.8, spike_factor:0.0, rotation_speed:0.4, pulse_amplitude:0.0, glow_intensity:0.0, listening_expansion:1.0, mode:"Companion".into(), speaking:false }
    }
}

fn emotion_params(emotion: &str) -> (f32, f32) {
    match emotion {
        "frustration" => (0.0, 0.9), "sadness" => (0.6, 0.0), "excitement" => (0.12, 0.0),
        "anxiety" => (0.75, 0.5), "confusion" => (0.0, 0.2), _ => (0.3, 0.0),
    }
}

fn draw_branch(x: f32, y: f32, length: f32, angle: f32, depth: u32, max_depth: u32, params: &VisualParams, time: f32, base_hue: f32, interactive: &InteractiveState) {
    if depth == 0 { return; }
    let end_x = x + length * angle.cos();
    let end_y = y + length * angle.sin();

    let depth_factor = depth as f32 / max_depth as f32;
    let pulse = if params.speaking { params.pulse_amplitude * (time * 8.0).sin() } else { 0.0 };
    let petting_pulse = interactive.petting_strength * 0.1 * (time * 4.0).sin();
    let lightness = (params.value + pulse * 0.1 + petting_pulse).clamp(0.3, 1.0);
    let saturation = (params.saturation + params.glow_intensity * 0.2 + interactive.hover_strength * 0.1).clamp(0.1, 1.0);
    let color = Color::from_hsl(params.hue, saturation, lightness);

    let thickness = (2.0 * (1.0 - depth_factor) + 1.0) * (1.0 + params.spike_factor * 0.5) * (1.0 + interactive.hover_strength * 0.2);
    draw_line(x, y, end_x, end_y, thickness, color);

    let new_length = length * 0.7;
    let angle_variance = match params.mode.as_str() {
        "Mentor" => PI / 5.0, "Trickster" => PI / 3.5, _ => PI / 4.5,
    };
    let spike_angle = params.spike_factor * PI / 6.0;
    draw_branch(end_x, end_y, new_length, angle - angle_variance - spike_angle, depth-1, max_depth, params, time, base_hue, interactive);
    draw_branch(end_x, end_y, new_length, angle + angle_variance + spike_angle, depth-1, max_depth, params, time, base_hue, interactive);
    if depth > 2 && max_depth > 4 {
        draw_branch(end_x, end_y, new_length * 0.6, angle, depth-1, max_depth, params, time, base_hue, interactive);
    }
}

#[macroquad::main("AI Companion Avatar")]
async fn main() {
    // Parse base hue from command line
    let base_hue: f32 = std::env::args()
        .collect::<Vec<_>>()
        .iter()
        .position(|a| a == "--hue")
        .and_then(|i| std::env::args().nth(i + 1))
        .and_then(|s| s.parse().ok())
        .unwrap_or(0.3);

    // Kill any existing process on our port (simple approach: just try to bind, if fails, exit old?)
    let listener = match TcpListener::bind("127.0.0.1:9001").await {
        Ok(l) => l,
        Err(_) => {
            // Port already in use – likely orphaned previous instance.
            // We'll just continue; the main app should have killed it.
            // For robustness, we could attempt to connect and send a shutdown command.
            eprintln!("Port 9001 already in use. Exiting.");
            return;
        }
    };
    println!("Interactive Avatar listening on 127.0.0.1:9001");

    // Accept a single connection (the main app)
    let (mut stream, _) = listener.accept().await.unwrap();
    // Send READY handshake
    stream.write_all(b"READY").await.unwrap();

    // Set up watch channel for AI state
    let (tx, mut rx) = watch::channel(AiState::default());
    let tx = Arc::new(tx);

    // Spawn task to read from TCP
    tokio::spawn(async move {
        let mut buf = vec![0u8; 1024];
        loop {
            match stream.read(&mut buf).await {
                Ok(n) if n > 0 => {
                    if let Ok(state) = serde_json::from_slice::<AiState>(&buf[..n]) {
                        let _ = tx.send(state);
                    }
                }
                _ => break,
            }
        }
    });

    let mut visual = VisualParams::default();
    let mut interactive = InteractiveState::new();
    let mut time = 0.0;
    let mut rotation = 0.0;
    let mut warning_flash = 0.0;
    let mut last_click_time = 0.0;
    let double_click_window = 0.3;

    let mut target_depth = 2.0;
    let mut target_hue = base_hue;
    let mut target_saturation = 0.5;
    let mut target_spike = 0.0;
    let mut target_speed = 0.4;
    let mut target_pulse = 0.0;

    // Window always on top initially to ensure visibility
    unsafe { macroquad::window::set_window_always_on_top(true); }
    // Release after a moment
    let mut top_timer = 0.0;

    loop {
        time = get_time() as f32;
        top_timer += 0.016;
        if top_timer > 2.0 {
            unsafe { macroquad::window::set_window_always_on_top(false); }
        }

        let mouse_pos = Vec2::new(mouse_position().0, mouse_position().1);
        let mouse_down = is_mouse_button_down(MouseButton::Left);
        let mouse_pressed = is_mouse_button_pressed(MouseButton::Left);

        let avatar_center = Vec2::new(screen_width()/2.0 + interactive.offset_x, screen_height()/2.0 + 30.0 + interactive.offset_y);
        let avatar_radius = 120.0;

        interactive.update(mouse_pos, mouse_down, mouse_pressed, avatar_center, avatar_radius);

        // Double-click detection
        if mouse_pressed && interactive.is_hovering {
            let now = time;
            if now - last_click_time < double_click_window {
                interactive.double_click_triggered = true;
                interactive.target_rotation_offset += PI * 2.0;
                for _ in 0..20 {
                    let angle = rand::gen_range(0.0, PI*2.0);
                    let speed = rand::gen_range(80.0, 200.0);
                    interactive.click_sparkles.push(Sparkle {
                        pos: avatar_center,
                        vel: Vec2::new(angle.cos()*speed, angle.sin()*speed),
                        life: 1.2,
                        color: Color::from_hsl(rand::gen_range(0.0, 1.0), 0.9, 0.8),
                    });
                }
            }
            last_click_time = now;
        }

        // Receive AI state (non‑blocking)
        if let Ok(state) = rx.borrow_and_update().map(|s| s.clone()) {
            let (emotion_shift, emotion_spike) = emotion_params(&state.emotion);
            target_hue = (base_hue + emotion_shift * 0.3).clamp(0.0, 1.0);
            target_saturation = 0.4 + state.trust * 0.6;
            target_depth = 2.0 + (state.message_count as f32 / 20.0).min(5.0);
            target_speed = match state.mode.as_str() {
                "Mentor" => 0.2, "Trickster" => 0.8, _ => 0.4,
            };
            if state.speaking { target_speed *= 1.8; target_pulse = 0.15; }
            else { target_pulse = 0.0; }
            target_spike = emotion_spike;
            visual.mode = state.mode;
            visual.speaking = state.speaking;

            if state.memory_glow { interactive.queue_memory_glow(); }
            if state.warning { warning_flash = 1.0; }
            visual.listening_expansion = if state.listening { 1.05 } else { 1.0 };
        }

        // Smooth interpolation
        visual.depth += (target_depth - visual.depth) * 0.05;
        visual.hue += (target_hue - visual.hue) * 0.05;
        visual.saturation += (target_saturation - visual.saturation) * 0.03;
        visual.spike_factor += (target_spike - visual.spike_factor) * 0.08;
        visual.rotation_speed += (target_speed - visual.rotation_speed) * 0.05;
        visual.pulse_amplitude += (target_pulse - visual.pulse_amplitude) * 0.1;

        warning_flash *= 0.9;
        visual.glow_intensity = interactive.memory_glow_queue + interactive.click_pulse * 0.3 + interactive.petting_strength * 0.2;

        rotation += (visual.rotation_speed + interactive.rotation_offset.abs() * 0.5) * 0.016;
        if interactive.double_click_triggered { rotation += 0.5; interactive.double_click_triggered = false; }

        // Render
        clear_background(Color::new(0.05, 0.05, 0.08, 1.0));

        if warning_flash > 0.01 {
            draw_rectangle(0.0, 0.0, screen_width(), screen_height(), Color::new(1.0, 0.0, 0.0, warning_flash * 0.3));
        }

        let scale = visual.listening_expansion * (1.0 + interactive.hover_strength * 0.03) * (1.0 + interactive.click_pulse * 0.05);
        let trunk_length = 100.0 * scale;
        draw_branch(avatar_center.x, avatar_center.y, trunk_length, -PI/2.0 + rotation * 0.1 + interactive.rotation_offset,
                    visual.depth.round() as u32, visual.depth.round() as u32, &visual, time, base_hue, &interactive);

        for s in &interactive.click_sparkles {
            draw_circle(s.pos.x, s.pos.y, 3.0 * s.life, s.color);
        }

        let heart_color = Color::from_hsl(visual.hue, visual.saturation, 0.9);
        draw_circle(avatar_center.x, avatar_center.y, 6.0 + visual.glow_intensity * 10.0, heart_color);

        if interactive.is_hovering && !interactive.is_dragging {
            draw_circle_lines(avatar_center.x, avatar_center.y, avatar_radius * 1.1, 1.5, Color::new(1.0, 1.0, 1.0, interactive.hover_strength * 0.5));
        }

        next_frame().await;
    }
}
```

```rust
use serde::{Deserialize, Serialize};
use std::io::{Read, Write};
use std::net::TcpStream;
use std::process::{Child, Command};
use std::sync::{Arc, Mutex};
use std::time::Duration;
use tokio::sync::mpsc;
use crate::domain::AvatarColor;

#[derive(Debug, Serialize, Deserialize, Clone)]
pub struct AvatarState {
    pub emotion: String,
    pub trust: f32,
    pub mode: String,
    pub speaking: bool,
    pub message_count: u32,
    pub memory_glow: bool,
    pub listening: bool,
    pub warning: bool,
}

enum AvatarCommand {
    SendState(AvatarState),
    Shutdown,
}

pub struct AvatarClient {
    sender: mpsc::UnboundedSender<AvatarCommand>,
    child: Arc<Mutex<Option<Child>>>,
}

impl AvatarClient {
    pub fn new(color: &AvatarColor) -> Self {
        let hue = color.to_hue();
        
        // Kill any existing avatar process on port 9001 (platform specific)
        #[cfg(target_os = "windows")]
        {
            let _ = std::process::Command::new("cmd")
                .args(["/C", "netstat -ano | findstr :9001"])
                .output();
        }
        #[cfg(not(target_os = "windows"))]
        {
            let _ = std::process::Command::new("sh")
                .arg("-c")
                .arg("lsof -ti:9001 | xargs kill -9")
                .output();
        }

        // Locate avatar binary relative to current executable
        let exe_dir = std::env::current_exe()
            .ok()
            .and_then(|p| p.parent().map(|p| p.to_path_buf()))
            .unwrap_or_else(|| std::env::current_dir().unwrap());
        
        let avatar_name = if cfg!(target_os = "windows") {
            "ai_companion_avatar.exe"
        } else {
            "ai_companion_avatar"
        };
        let avatar_path = exe_dir.join("avatar").join(avatar_name);

        let mut child = Command::new(&avatar_path)
            .arg("--hue")
            .arg(hue.to_string())
            .spawn()
            .expect("Failed to start avatar process");

        // Wait for TCP connection and READY handshake
        let mut connected = false;
        for _ in 0..20 {
            std::thread::sleep(Duration::from_millis(100));
            if let Ok(mut stream) = TcpStream::connect("127.0.0.1:9001") {
                let mut buf = [0u8; 5];
                if stream.read_exact(&mut buf).is_ok() && &buf == b"READY" {
                    connected = true;
                    break;
                }
            }
        }
        if !connected {
            log::error!("Avatar process started but never sent READY");
        }

        let (sender, mut receiver) = mpsc::unbounded_channel::<AvatarCommand>();
        let child_handle = Arc::new(Mutex::new(Some(child)));

        // Background task to send state updates (throttled to 30Hz)
        tokio::spawn(async move {
            let mut stream: Option<TcpStream> = None;
            let mut last_send = std::time::Instant::now();
            let mut pending_state: Option<AvatarState> = None;

            while let Some(cmd) = receiver.recv().await {
                match cmd {
                    AvatarCommand::SendState(state) => {
                        pending_state = Some(state);
                    }
                    AvatarCommand::Shutdown => break,
                }

                // Throttle to ~30Hz
                let now = std::time::Instant::now();
                if now.duration_since(last_send) >= Duration::from_millis(33) {
                    if let Some(ref state) = pending_state {
                        if stream.is_none() {
                            stream = TcpStream::connect("127.0.0.1:9001").ok();
                        }
                        if let Some(s) = stream.as_mut() {
                            let json = serde_json::to_string(state).unwrap();
                            if s.write_all(json.as_bytes()).is_err() {
                                stream = None;
                            }
                        }
                        pending_state = None;
                        last_send = now;
                    }
                }
            }
        });

        Self { sender, child: child_handle }
    }

    pub fn send(&self, state: &AvatarState) {
        let _ = self.sender.send(AvatarCommand::SendState(state.clone()));
    }

    pub fn update_color(&self, _color: &AvatarColor) {
        // For simplicity, restart required; or we could send a special command.
        log::info!("Avatar color change requested – requires restart.");
    }
}

impl Drop for AvatarClient {
    fn drop(&mut self) {
        let _ = self.sender.send(AvatarCommand::Shutdown);
        if let Ok(mut child_guard) = self.child.lock() {
            if let Some(mut child) = child_guard.take() {
                let _ = child.kill();
            }
        }
    }
}
```
