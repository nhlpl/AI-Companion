## 🖥️ `avatar/src/main.rs` (Complete)

```rust
use macroquad::prelude::*;
use serde::{Deserialize, Serialize};
use std::collections::VecDeque;
use std::f32::consts::PI;
use std::sync::Arc;
use tokio::net::TcpListener;
use tokio::sync::watch;

// ---------- State received from main app ----------
#[derive(Debug, Deserialize, Serialize, Clone)]
struct AvatarState {
    emotion: String,          // "", "frustration", "sadness", "excitement", "anxiety", "confusion"
    trust: f32,               // 0.15 – 0.98
    mode: String,             // "Mentor", "Companion", "Trickster"
    speaking: bool,           // true while AI generates response
    message_count: u32,       // total messages exchanged
    memory_glow: bool,        // pulsed true when memory retrieved
    listening: bool,          // VAD from main app (or computed locally)
    warning: bool,            // rare boundary alert
}

impl Default for AvatarState {
    fn default() -> Self {
        Self {
            emotion: String::new(),
            trust: 0.5,
            mode: "Companion".to_string(),
            speaking: false,
            message_count: 0,
            memory_glow: false,
            listening: false,
            warning: false,
        }
    }
}

// ---------- Visual parameters (smoothly interpolated) ----------
struct VisualParams {
    depth: f32,               // fractal recursion depth (smoothed)
    hue: f32,
    saturation: f32,
    value: f32,
    spike_factor: f32,
    rotation_speed: f32,
    pulse_amplitude: f32,
    glow_intensity: f32,
    listening_expansion: f32,
}

impl Default for VisualParams {
    fn default() -> Self {
        Self {
            depth: 2.0,
            hue: 0.3,
            saturation: 0.5,
            value: 0.8,
            spike_factor: 0.0,
            rotation_speed: 0.4,
            pulse_amplitude: 0.0,
            glow_intensity: 0.0,
            listening_expansion: 1.0,
        }
    }
}

// ---------- Helper: map emotion to hue shift and spike factor ----------
fn emotion_params(emotion: &str) -> (f32, f32) {
    match emotion {
        "frustration" => (0.0, 0.9),   // red, spiky
        "sadness" => (0.6, 0.0),       // blue, soft droop
        "excitement" => (0.12, 0.0),   // yellow-orange, bouncy
        "anxiety" => (0.75, 0.5),      // purple, jittery
        "confusion" => (0.0, 0.2),     // gray, irregular
        _ => (0.3, 0.0),               // green, calm
    }
}

// ---------- Fractal drawing ----------
fn draw_branch(
    x: f32,
    y: f32,
    length: f32,
    angle: f32,
    depth: u32,
    max_depth: u32,
    params: &VisualParams,
    time: f32,
    base_hue: f32,
) {
    if depth == 0 {
        return;
    }

    let end_x = x + length * angle.cos();
    let end_y = y + length * angle.sin();

    // Color based on depth and current state
    let depth_factor = depth as f32 / max_depth as f32;
    let pulse = if params.speaking {
        params.pulse_amplitude * (time * 8.0).sin()
    } else {
        0.0
    };
    let lightness = (params.value + pulse * 0.1).clamp(0.3, 1.0);
    let saturation = (params.saturation + params.glow_intensity * 0.2).clamp(0.1, 1.0);
    let color = Color::from_hsl(params.hue, saturation, lightness);

    // Thickness decreases with depth, increases with spike factor at tips
    let thickness = (2.0 * (1.0 - depth_factor) + 1.0) * (1.0 + params.spike_factor * 0.5);
    draw_line(x, y, end_x, end_y, thickness, color);

    // Recursive branches
    let new_length = length * 0.7;
    let angle_variance = match params.mode.as_str() {
        "Mentor" => PI / 5.0,      // 36° – structured
        "Trickster" => PI / 3.5,   // ~51° – wide, playful
        _ => PI / 4.5,             // 40° – balanced
    };
    let spike_angle = params.spike_factor * PI / 6.0;
    let left_angle = angle - angle_variance - spike_angle;
    let right_angle = angle + angle_variance + spike_angle;

    draw_branch(end_x, end_y, new_length, left_angle, depth - 1, max_depth, params, time, base_hue);
    draw_branch(end_x, end_y, new_length, right_angle, depth - 1, max_depth, params, time, base_hue);

    // Occasionally add a third branch for complexity at higher depths
    if depth > 2 && max_depth > 4 {
        draw_branch(end_x, end_y, new_length * 0.6, angle, depth - 1, max_depth, params, time, base_hue);
    }
}

// ---------- Audio input for listening detection (simple VAD) ----------
fn setup_audio() -> Option<cpal::Stream> {
    use cpal::traits::{DeviceTrait, HostTrait, StreamTrait};
    let host = cpal::default_host();
    let device = host.default_input_device()?;
    let config = device.default_input_config().ok()?;
    let sample_rate = config.sample_rate().0;

    let (tx, mut rx) = tokio::sync::mpsc::channel::<f32>(32);
    let stream = device.build_input_stream(
        &config.into(),
        move |data: &[f32], _: &cpal::InputCallbackInfo| {
            let rms = (data.iter().map(|s| s * s).sum::<f32>() / data.len() as f32).sqrt();
            let _ = tx.try_send(rms);
        },
        |err| eprintln!("Audio error: {}", err),
        None,
    ).ok()?;
    stream.play().ok()?;

    // Spawn task to update listening state
    tokio::spawn(async move {
        let mut vad_state = false;
        let threshold = 0.02;
        while let Some(rms) = rx.recv().await {
            let speaking = rms > threshold;
            if speaking != vad_state {
                vad_state = speaking;
                // Could send to main loop via watch channel
            }
        }
    });
    Some(stream)
}

// ---------- Main ----------
#[macroquad::main("AI Companion Avatar")]
async fn main() {
    // Parse base hue from command line (e.g., --hue 0.75)
    let base_hue: f32 = std::env::args()
        .collect::<Vec<_>>()
        .iter()
        .position(|a| a == "--hue")
        .and_then(|i| std::env::args().nth(i + 1))
        .and_then(|s| s.parse().ok())
        .unwrap_or(0.3); // default Forest

    // TCP server for state updates
    let listener = TcpListener::bind("127.0.0.1:9001").await.unwrap();
    println!("Avatar listening on 127.0.0.1:9001 with base hue {:.2}", base_hue);

    // Watch channel for state
    let (tx, mut rx) = watch::channel(AvatarState::default());
    let tx = Arc::new(tx);

    // Spawn connection handler
    tokio::spawn(async move {
        loop {
            if let Ok((mut stream, _)) = listener.accept().await {
                let tx = tx.clone();
                tokio::spawn(async move {
                    let mut buf = vec![0u8; 1024];
                    loop {
                        match stream.try_read(&mut buf) {
                            Ok(n) if n > 0 => {
                                if let Ok(state) = serde_json::from_slice::<AvatarState>(&buf[..n]) {
                                    let _ = tx.send(state);
                                }
                            }
                            _ => break,
                        }
                    }
                });
            }
        }
    });

    // Setup audio (optional)
    let _audio_stream = setup_audio();

    // Visual state
    let mut visual = VisualParams::default();
    let mut time = 0.0;
    let mut rotation = 0.0;
    let mut memory_glow_decay = 0.0;
    let mut warning_flash = 0.0;
    let mut pulse_phase = 0.0;

    // For smooth interpolation
    let mut target_depth = 2.0;
    let mut target_hue = base_hue;
    let mut target_saturation = 0.5;
    let mut target_spike = 0.0;
    let mut target_speed = 0.4;
    let mut target_pulse = 0.0;

    loop {
        time = get_time() as f32;
        pulse_phase += 0.05;

        // Get latest state
        if let Ok(state) = rx.borrow_and_update().map(|s| s.clone()) {
            // Update targets
            let (emotion_hue_shift, emotion_spike) = emotion_params(&state.emotion);
            target_hue = (base_hue + emotion_hue_shift * 0.3).clamp(0.0, 1.0);
            target_saturation = 0.4 + state.trust * 0.6;
            target_depth = 2.0 + (state.message_count as f32 / 20.0).min(5.0);
            target_speed = match state.mode.as_str() {
                "Mentor" => 0.2,
                "Trickster" => 0.8,
                _ => 0.4,
            };
            if state.speaking {
                target_speed *= 1.8;
                target_pulse = 0.15;
            } else {
                target_pulse = 0.0;
            }
            target_spike = emotion_spike;
            visual.mode = state.mode.clone();
            visual.speaking = state.speaking;

            if state.memory_glow {
                memory_glow_decay = 1.0;
            }
            if state.warning {
                warning_flash = 1.0;
            }
            // Listening expansion (could also come from local VAD)
            visual.listening_expansion = if state.listening { 1.05 } else { 1.0 };
        }

        // Smooth interpolation
        visual.depth += (target_depth - visual.depth) * 0.05;
        visual.hue += (target_hue - visual.hue) * 0.05;
        visual.saturation += (target_saturation - visual.saturation) * 0.03;
        visual.spike_factor += (target_spike - visual.spike_factor) * 0.08;
        visual.rotation_speed += (target_speed - visual.rotation_speed) * 0.05;
        visual.pulse_amplitude += (target_pulse - visual.pulse_amplitude) * 0.1;

        // Decay transient effects
        memory_glow_decay *= 0.95;
        warning_flash *= 0.9;
        visual.glow_intensity = memory_glow_decay;

        // Update rotation
        rotation += visual.rotation_speed * 0.016;

        // Render
        clear_background(Color::new(0.05, 0.05, 0.08, 1.0)); // deep space

        let center_x = screen_width() / 2.0;
        let center_y = screen_height() / 2.0 + 30.0;

        // Listening expansion
        let scale = visual.listening_expansion * (1.0 + 0.02 * (time * 2.0).sin());

        // Warning flash overlay
        if warning_flash > 0.01 {
            draw_rectangle(0.0, 0.0, screen_width(), screen_height(), Color::new(1.0, 0.0, 0.0, warning_flash * 0.3));
        }

        // Draw fractal
        let trunk_length = 100.0 * scale;
        draw_branch(
            center_x,
            center_y,
            trunk_length,
            -PI / 2.0 + rotation * 0.1, // subtle rotation
            visual.depth.round() as u32,
            visual.depth.round() as u32,
            &visual,
            time,
            base_hue,
        );

        // Draw a soft central glow (the "heart")
        let heart_color = Color::from_hsl(visual.hue, visual.saturation, 0.9);
        draw_circle(center_x, center_y, 6.0 + memory_glow_decay * 10.0, heart_color);

        next_frame().await;
    }
}
```

## ✅ Summary

This complete living avatar:
- Accepts a base color via `--hue` (0.0–1.0) from the main app.
- Receives real-time state updates over TCP.
- Renders a fractal tree that evolves with message count.
- Shifts hue based on emotion, saturation on trust.
- Changes shape regularity based on mode.
- Pulses when speaking, expands when listening.
- Glows warmly when retrieving memories.
- Flashes red for warnings.

It's a beautiful, silent witness to the user's journey with DeepSeek—fully embodying the psychological principles of a trustworthy, warm, and living companion.
