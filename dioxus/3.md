fn text_hash(text: &str) -> u64 {
    let mut hasher = AHasher::default();
    text.hash(&mut hasher);
    hasher.finish()
}

fn similarity(h1: u64, h2: u64) -> f32 {
    let diff = (h1 ^ h2).count_ones();
    1.0 - (diff as f32 / 64.0)
}

// ------------------------------
// Bloom filter for deduplication
// ------------------------------
#[derive(Serialize, Deserialize, Clone)]
struct BloomFilter {
    bits: Vec<u64>,
    ring: Vec<u64>,
    pos: usize,
    capacity: usize,
}

impl BloomFilter {
    fn new(capacity: usize) -> Self {
        let words = (BLOOM_SIZE + 63) / 64;
        Self { bits: vec![0; words], ring: vec![0; capacity], pos: 0, capacity }
    }

    fn hash_pair(h: u64) -> (usize, usize) {
        let h1 = h.wrapping_mul(0x9e3779b97f4a7c15);
        let h2 = h.wrapping_mul(0xbf58476d1ce4e5b9);
        (h1 as usize % BLOOM_SIZE, h2 as usize % BLOOM_SIZE)
    }

    fn insert(&mut self, hash: u64) {
        let (i1, i2) = Self::hash_pair(hash);
        self.bits[i1 / 64] |= 1 << (i1 % 64);
        self.bits[i2 / 64] |= 1 << (i2 % 64);
        self.ring[self.pos] = hash;
        self.pos = (self.pos + 1) % self.capacity;
    }

    fn exact_duplicate(&self, hash: u64) -> bool {
        self.ring.contains(&hash)
    }
}

// ------------------------------
// Inverted index (8 bytes, 256 buckets each)
// ------------------------------
#[derive(Serialize, Deserialize, Clone)]
struct InvertedIndex {
    buckets: [Vec<Vec<usize>>; 8],
}

impl InvertedIndex {
    fn new() -> Self {
        const EMPTY: Vec<Vec<usize>> = Vec::new();
        let mut buckets = [EMPTY; 8];
        for i in 0..8 {
            buckets[i] = vec![Vec::new(); 256];
        }
        Self { buckets }
    }

    fn clear(&mut self) {
        for pos in 0..8 {
            for byte in 0..256 {
                self.buckets[pos][byte].clear();
            }
        }
    }

    fn insert(&mut self, hash: u64, idx: usize) {
        let bytes = hash.to_le_bytes();
        for (pos, &byte) in bytes.iter().enumerate() {
            self.buckets[pos][byte as usize].push(idx);
        }
    }

    fn build(&mut self, memories: &[Memory]) {
        self.clear();
        for (idx, mem) in memories.iter().enumerate() {
            self.insert(mem.hash, idx);
        }
    }

    fn query(&self, query_hash: u64) -> Vec<usize> {
        let bytes = query_hash.to_le_bytes();
        let mut candidates: Vec<usize> = Vec::new();
        let mut seen = HashSet::new();

        let mut pos_sizes: Vec<_> = (0..4)
            .map(|pos| (pos, self.buckets[pos][bytes[pos] as usize].len()))
            .collect();
        pos_sizes.sort_by_key(|&(_, sz)| sz);

        for (pos, _) in pos_sizes.iter().take(2) {
            for &idx in &self.buckets[*pos][bytes[*pos] as usize] {
                if seen.insert(idx) {
                    candidates.push(idx);
                }
            }
        }
        candidates
    }
}

#[derive(Debug, Clone, Copy, PartialEq, Eq)]

    importance: u16,
    id: usize,
}

impl Ord for ImportanceEntry {
    fn cmp(&self, other: &Self) -> std::cmp::Ordering {
        self.importance.cmp(&other.importance).then(self.id.cmp(&other.id))
    }
}

impl PartialOrd for ImportanceEntry {
    fn partial_cmp(&self, other: &Self) -> Option<std::cmp::Ordering> {
        Some(self.cmp(other))
    }
}

#[derive(Clone)]
struct ConversationState {
    nodes: [bool; NUM_STATE_NODES],
}

impl ConversationState {
    fn new() -> Self {
        Self { nodes: [false; NUM_STATE_NODES] }
    }

    fn update(&mut self, weights: &[[f32; NUM_STATE_NODES]; NUM_STATE_NODES], biases: &[f32; NUM_STATE_NODES]) {
        let mut new_nodes = [false; NUM_STATE_NODES];
        for i in 0..NUM_STATE_NODES {
            let sum: f32 = (0..NUM_STATE_NODES).map(|j| {
                if self.nodes[j] { weights[i][j] } else { 0.0 }
            }).sum::<f32>() + biases[i];
            new_nodes[i] = sum > 0.0;
        }
        self.nodes = new_nodes;
    }

    fn to_mode(&self) -> ConversationMode {
        // Map binary state to one of 4 modes (attractors)
        let idx = (if self.nodes[0] { 1 } else { 0 }) << 3 |
                  (if self.nodes[1] { 1 } else { 0 }) << 2 |
                  (if self.nodes[2] { 1 } else { 0 }) << 1 |
                  (if self.nodes[3] { 1 } else { 0 });
        match idx {
            0..=3 => ConversationMode::Curious,
            4..=7 => ConversationMode::Analytical,
            8..=11 => ConversationMode::Playful,
            _ => ConversationMode::Formal,
        }
    }
}

    // Bio: Power-law forgetting curve

    fn power_law_recency(&self, days: f32) -> f32 {
        (1.0 + FORGETTING_BETA * days).powf(-FORGETTING_ALPHA)

    fn harmony_score(&self, mem: &Memory, sim: f32, current_time: f64) -> f32 {
        if !self.settings.read().harmony_mode {
            return sim;
        }
        let days = ((current_time - mem.timestamp) / 86400.0) as f32;
        let recency = self.power_law_recency(days);
        let importance = mem.importance as f32 / 65535.0;
        let s = self.settings.read();
        s.similarity_weight * sim + s.importance_weight * importance + s.recency_weight * recency
    }

    // ------------------------------
    // Bio: STDP importance update
    // ------------------------------
    fn apply_stdp(&mut self, memory_id: usize, retrieval_time: f64, user_action_time: Option<f64>) {
        if let Some(action_time) = user_action_time {
            let dt = action_time - retrieval_time;
            let mut memories = self.memories.write();
            if let Some(mem) = memories.iter_mut().find(|m| m.id == memory_id) {
                let delta = if dt > 0.0 {
                    // pre (retrieval) before post (user action) -> potentiation
                    STDP_A_PLUS * (-dt / STDP_TAU_PLUS).exp()
                } else {
                    // post before pre -> depression
                    -STDP_A_MINUS * (dt / STDP_TAU_MINUS).exp()
                };
                let new_imp = (mem.importance as f32 + delta * 65535.0).clamp(0.0, 65535.0) as u16;
                if new_imp != mem.importance {
                    if !mem.permanent {
                        self.importance_set.write().remove(&ImportanceEntry { importance: mem.importance, id: mem.id });
                    }
                    mem.importance = new_imp;
                    if !mem.permanent {
                        self.importance_set.write().insert(ImportanceEntry { importance: new_imp, id: mem.id });
                    }
                }
            }
        }
    }

    fn clonal_selection_update(&mut self, positive_feedback: bool) {

        let target = if positive_feedback { 1.0 } else { 0.0 };
        let best = clones.into_iter().min_by(|a, b| {
            (a - target).abs().partial_cmp(&(b - target).abs()).unwrap()
        }).unwrap();
        let new_val = (best * 15.0).round() as u8;
        *self.personality.write() = new_val;
        self.trigger_save();
    }

    // ------------------------------
    // Bio: CMA-ES hyperparameter optimization (simplified)
    // ------------------------------
    fn cma_es_step(&mut self) {
        let mut rng = rand::thread_rng();
        let current_settings = self.settings.read().clone();
        // Sample population around current settings
        let mut candidates: Vec<(Settings, f32)> = (0..CMA_POP_SIZE).map(|_| {
            let mut s = current_settings.clone();
            let dist = Uniform::new(-CMA_SIGMA, CMA_SIGMA);
            s.recency_weight = (s.recency_weight + dist.sample(&mut rng)).clamp(0.0, 1.0);
            s.importance_weight = (s.importance_weight + dist.sample(&mut rng)).clamp(0.0, 1.0);
            s.similarity_weight = (s.similarity_weight + dist.sample(&mut rng)).clamp(0.0, 1.0);
            // Normalize weights
            let sum = s.recency_weight + s.importance_weight + s.similarity_weight;
            if sum > 0.0 {
                s.recency_weight /= sum;
                s.importance_weight /= sum;
                s.similarity_weight /= sum;
            }
            (s, 0.0) // fitness placeholder
        }).collect();

        // Evaluate fitness based on recent feedback (positive ratio)
        let fb = self.feedback_history.read();
        let recent: Vec<_> = fb.iter().rev().take(20).collect();
        let positive_ratio = if recent.is_empty() { 0.5 } else {
            recent.iter().filter(|e| e.positive).count() as f32 / recent.len() as f32
        };
        // Fitness = positive_ratio (want to maximize)
        for (s, fitness) in &mut candidates {
            *fitness = positive_ratio; // all same for now; could simulate with these settings
        }
        // Select best candidate (just pick the one with highest fitness; all equal, so pick first)
        // For simplicity, just keep current settings; in real CMA-ES we'd update mean and covariance.
        // We'll just add small random drift toward better.
        let drift = if positive_ratio > 0.6 { 0.01 } else if positive_ratio < 0.4 { -0.01 } else { 0.0 };
        if drift != 0.0 {
            let mut s = self.settings.write();
            s.recency_weight = (s.recency_weight + drift).clamp(0.1, 0.9);
            s.importance_weight = (s.importance_weight + drift).clamp(0.1, 0.9);
            s.similarity_weight = (s.similarity_weight - drift).clamp(0.1, 0.9);
            // renormalize
            let sum = s.recency_weight + s.importance_weight + s.similarity_weight;
            s.recency_weight /= sum;
            s.importance_weight /= sum;
            s.similarity_weight /= sum;
        }
    }

            }

        let weights = self.state_weights.read();
        let biases = self.state_biases.read();
        state.update(&weights, &biases);
    }

    // ------------------------------
    // Core memory operations (with bio features)

    fn add_memory(&mut self, text: String, importance_f32: f32, permanent: bool) -> usize {
        let importance = (importance_f32 * 65535.0) as u16;
        let hash = text_hash(&text);

        if self.bloom.read().exact_duplicate(hash) {
            let mut memories = self.memories.write();
            if let Some(pos) = memories.iter().position(|m| m.hash == hash && m.text == text) {
                let mem = &mut memories[pos];
                if importance > mem.importance {
                    if !mem.permanent {
                        self.importance_set.write().remove(&ImportanceEntry { importance: mem.importance, id: mem.id });
                    }
                    mem.importance = importance;
                    mem.timestamp = Self::now_secs();
                    mem.permanent = permanent;
                    if !permanent {
                        self.importance_set.write().insert(ImportanceEntry { importance, id: mem.id });
                    }
                    self.index_dirty.set(true);
                }
                self.trigger_save();
                return mem.id;
            }
        }

        let id = *self.next_id.read();
        *self.next_id.write() += 1;

        let mem = Memory {
            id, text: text.clone(), hash, timestamp: Self::now_secs(),
            importance, permanent, last_retrieved: None, retrieval_count: 0,
        };

        self.bloom.write().insert(hash);
        self.memories.write().push(mem.clone());
        self.inverted_index.write().insert(hash, self.memories.read().len() - 1);
        if !permanent {
            self.importance_set.write().insert(ImportanceEntry { importance, id });
        }
        self.index_dirty.set(true);

        if self.memories.read().len() > MAX_MEM {
            self.prune_low_importance();
        }

        self.trigger_save();
        id
    }

    fn prune_low_importance(&mut self) {
        let mut memories = self.memories.write();
        let mut importance_set = self.importance_set.write();
        while memories.len() > MAX_MEM {
            if let Some(entry) = importance_set.first().copied() {
                importance_set.remove(&entry);
                if let Some(pos) = memories.iter().position(|m| m.id == entry.id && !m.permanent) {
                    memories.swap_remove(pos);
                }
            } else {
                break;
            }
        }
        self.index_dirty.set(true);
    }

    fn retrieve(&mut self, query: &str, top_k: usize) -> Vec<String> {
        let current_time = Self::now_secs();
        let query_hash = text_hash(query);

        if let Some(cached) = self.cache.read().get(&query_hash) {
            let memories = self.memories.read();
            return cached.iter()
                .filter_map(|&id| memories.iter().find(|m| m.id == id))
                .take(top_k)
                .map(|m| m.text.clone())
                .collect();
        }

        if *self.index_dirty.read() {
            self.inverted_index.write().build(&self.memories.read());
            self.index_dirty.set(false);
        }

        let index = self.inverted_index.read();
        let candidates = index.query(query_hash);
        let memories = self.memories.read();

        let mut scored: Vec<(f32, usize)> = candidates.into_iter()
            .map(|idx| {
                let mem = &memories[idx];
                let sim = similarity(query_hash, mem.hash);
                let score = self.harmony_score(mem, sim, current_time);
                (score, idx)
            })
            .collect();

        scored.sort_by(|a, b| b.0.partial_cmp(&a.0).unwrap());

        // Serendipity (with optimized chance)
        if self.settings.read().harmony_mode && rand::thread_rng().gen_bool(self.settings.read().serendipity_chance) {
            let low_imp: Vec<_> = memories.iter().enumerate()
                .filter(|(_, m)| m.importance < PRUNE_THRESHOLD && !m.permanent)
                .collect();
            if let Some(&(rand_idx, _)) = low_imp.get(rand::thread_rng().gen_range(0..low_imp.len())) {
                scored.insert(1, (0.8, rand_idx));
            }
        }

        let mut results = Vec::new();
        let mut ids = Vec::new();
        let threshold = self.settings.read().early_stop_threshold;
        for (score, idx) in scored {
            if results.len() >= top_k && score > (1.0 - threshold) {
                break;
            }
            results.push(memories[idx].text.clone());
            ids.push(memories[idx].id);
        }

        // Record retrieval for STDP
        let retrieval_time = current_time;
        let mut history = self.retrieval_history.write();
        for &id in &ids {
            history.push_back((id, retrieval_time));
            if history.len() > 100 { history.pop_front(); }
            // Update last_retrieved in memory
            if let Some(mut memories) = self.memories.try_write() {
                if let Some(mem) = memories.iter_mut().find(|m| m.id == id) {
                    mem.last_retrieved = Some(retrieval_time);
                    mem.retrieval_count += 1;
                }
            }
        }

        self.cache.write().put(query_hash, ids);
        results
    }

    fn add_anti_memory(&mut self, incorrect: String, correct: String) {
        self.anti_memories.write().push(AntiMemory {
            incorrect, correct, successes: 1, failures: 1,
            created: Self::now_secs(), strength: 1.0,
        });
        self.trigger_save();
    }

    fn get_warnings(&self) -> Vec<String> {
        self.anti_memories.read().iter()
            .filter(|a| a.strength > 0.5)
            .map(|a| format!("Do NOT say '{}'. Instead say '{}'.", a.incorrect, a.correct))
            .collect()
    }
}

#[component]
fn Chat() -> Element {
    let mut state = use_context::<AppState>();
    let input_text = state.input_text;
    let is_loading = state.is_loading;
    let messages = state.chat_messages;
    let visible_start = state.visible_start;

    use_effect(move || {
        let _ = state.save_trigger.read();
        let persisted = use_storage::<PersistedState>("ai_companion".to_string(), PersistedState::default);
        spawn(async move {
            gloo_timers::future::TimeoutFuture::new(SAVE_DEBOUNCE_MS).await;
            let mut p = persisted.write();
            p.memories = state.memories.read().clone();
            p.anti_memories = state.anti_memories.read().clone();
            p.personality = *state.personality.read();
            p.settings = state.settings.read().clone();
            p.next_id = *state.next_id.read();
            p.state_weights = *state.state_weights.read();
            p.state_biases = *state.state_biases.read();
        });
    });

    let send_message = move |_| {
        let text = input_text.read().clone();
        if text.trim().is_empty() || *is_loading.read() {
            return;
        }
        *is_loading.write() = true;

        // Update conversation state with user input
        state.update_conversation_state(&text);

        let mut msgs = messages.write();
        let msg_id = msgs.len();
        msgs.push(ChatMessage { role: "user".to_string(), content: text.clone(), id: msg_id });
        *input_text.write() = String::new();
        *visible_start.write() = msgs.len().saturating_sub(VISIBLE_MESSAGES);

        // Store user message as memory
        state.add_memory(format!("User: {}", text), 0.6, false);

        // Record response start time for STDP
        let response_start = AppState::now_secs();
        *state.last_response_time.write() = Some(response_start);

        spawn(async move {
            // Simulate thinking
            tokio::time::sleep(std::time::Duration::from_millis(800)).await;

            // Generate response using bio-inspired retrieval
            let raw_response = state.generate_response(&text);
            let filtered = state.filter_response(&raw_response);
            let final_response = if filtered.is_empty() { raw_response } else { filtered };

            // Add AI response to chat
            let ai_msg_id = state.chat_messages.read().len();
            state.chat_messages.write().push(ChatMessage {
                role: "ai".to_string(),
                content: final_response.clone(),
                id: ai_msg_id,
            });

            // Store AI response as memory
            state.add_memory(format!("AI: {}", final_response), 0.5, false);

            *state.visible_start.write() = state.chat_messages.read().len().saturating_sub(VISIBLE_MESSAGES);
            *is_loading.write() = false;

            // Apply STDP to recently retrieved memories (user action = AI response generated)
            let action_time = AppState::now_secs();
            let retrieval_history = state.retrieval_history.read().clone();
            for (mem_id, ret_time) in retrieval_history.iter() {
                // Only update if retrieval was within last 30 seconds
                if action_time - ret_time < 30.0 {
                    state.apply_stdp(*mem_id, *ret_time, Some(action_time));
                }
            }

            // Run CMA-ES step periodically
            if rand::thread_rng().gen_bool(0.1) {
                state.cma_es_step();
            }
        });
    };

    let load_earlier = move |_| {
        *visible_start.write() = visible_start.read().saturating_sub(VISIBLE_MESSAGES);
    };

    let total_messages = messages.read().len();
    let start_idx = *visible_start.read();
    let end_idx = (start_idx + VISIBLE_MESSAGES).min(total_messages);
    let can_load_more = start_idx > 0;

    let visible_messages: Vec<ChatMessage> = if total_messages > 0 {
        messages.read()[start_idx..end_idx].to_vec()
    } else {
        Vec::new()
    };

    // Feedback handlers
    let handle_positive = move |msg_id: usize| {
        let mut feedback = state.feedback_history.write();
        feedback.push_back(FeedbackEvent {
            timestamp: AppState::now_secs(),
            positive: true,
            personality_before: *state.personality.read(),
            settings_before: state.settings.read().clone(),
        });
        if feedback.len() > 50 { feedback.pop_front(); }
        state.clonal_selection_update(true);
    };

    let handle_negative = move |msg_id: usize| {
        let mut feedback = state.feedback_history.write();
        feedback.push_back(FeedbackEvent {
            timestamp: AppState::now_secs(),
            positive: false,
            personality_before: *state.personality.read(),
            settings_before: state.settings.read().clone(),
        });
        if feedback.len() > 50 { feedback.pop_front(); }
        state.clonal_selection_update(false);
    };

    rsx! {
        div { class: "flex flex-col h-full max-w-3xl mx-auto",
            div { class: "flex-1 overflow-y-auto mb-4 space-y-3",
                if can_load_more {
                    div { class: "text-center",
                        button {
                            class: "px-4 py-2 bg-gray-700 rounded-full text-sm hover:bg-gray-600 transition",
                            onclick: load_earlier,
                            "↑ Load earlier"
                        }
                    }
                }
                for msg in visible_messages {
                    div { class: "space-y-1",
                        div {
                            class: if msg.role == "user" {
                                "bg-blue-600 self-end max-w-[80%] p-3 rounded-2xl ml-auto"
                            } else {
                                "bg-gray-700 self-start max-w-[80%] p-3 rounded-2xl"
                            },
                            strong { "{if msg.role == \"user\" { \"You\" } else { \"AI\" }}: " }
                            "{msg.content}"
                        }
                        if msg.role == "ai" {
                            div { class: "flex gap-2 ml-2",
                                button {
                                    class: "text-xs px-2 py-1 bg-green-600/30 rounded hover:bg-green-600",
                                    onclick: move |_| handle_positive(msg.id),
                                    "👍"
                                }
                                button {
                                    class: "text-xs px-2 py-1 bg-red-600/30 rounded hover:bg-red-600",
                                    onclick: move |_| handle_negative(msg.id),
                                    "👎"
                                }
                            }
                        }
                    }
                }
                if *is_loading.read() {
                    div { class: "bg-gray-700 self-start max-w-[80%] p-3 rounded-2xl",
                        strong { "AI: " }
                        span { class: "inline-block w-5 h-5 border-2 border-gray-400 border-t-white rounded-full animate-spin ml-2" }
                    }
                }
            }
            div { class: "flex gap-3",
                input {
                    class: "flex-1 px-4 py-3 bg-gray-700 border-none rounded-full text-white focus:ring-2 focus:ring-blue-500 outline-none",
                    placeholder: "Type a message...",
                    value: "{input_text}",
                    oninput: move |e| input_text.set(e.value()),
                    onkeydown: move |e| if e.key() == Key::Enter { send_message(()) },
                }
                button {
                    class: "px-6 py-3 bg-blue-600 rounded-full font-bold hover:bg-blue-700 transition disabled:opacity-50",
                    onclick: send_message,
                    disabled: *is_loading.read(),
                    "Send"
                }
            }
        }
    }
}

#[component]
fn Settings() -> Element {
    let mut state = use_context::<AppState>();
    let personality = state.personality;
    let settings = state.settings;
    let mut anti_incorrect = use_signal(String::new);
    let mut anti_correct = use_signal(String::new);

        let new = (*personality.read() as i16 + delta as i16).clamp(0, 15);

        let inc = anti_incorrect.read().clone();

            *anti_incorrect.write() = String::new();
            *anti_correct.write() = String::new();

        div { class: "max-w-lg mx-auto space-y-6",

            div { class: "space-y-2",
                h3 { class: "text-lg" "Personality (Hex: {format!(\"{:X}\", personality.read())})" }

                    button { class: "px-4 py-2 bg-blue-600 rounded", onclick: move |_| adjust_personality(1), "+ Intensity" }

            hr { class: "border-gray-700" }

                h3 { class: "text-lg" "Anti‑Memory (Negative Selection)" }

                    class: "w-full px-4 py-2 bg-gray-700 rounded-full text-white",
                    placeholder: "Correct phrase",
                    value: "{anti_correct}",
                    oninput: move |e| anti_correct.set(e.value()),
                }
                button { class: "px-4 py-2 bg-blue-600 rounded-full", onclick: add_anti, "Add Correction" }
            }
        }
    }
