### Trust Calculation with Smoothing and Decay

```rust
pub fn record_feedback(&mut self, positive: bool) {
    let now = Self::now_secs();
    // Decay old feedback (half-life 7 days)
    let mut history = self.feedback_history.write();
    history.retain(|(ts, _)| now - ts < 7.0 * 86400.0);
    history.push_back((now, positive));
    if history.len() > 50 { history.pop_front(); }

    // Bayesian smoothing: (positives + 1) / (total + 2)
    let pos = history.iter().filter(|(_, p)| *p).count() as f32;
    let total = history.len() as f32;
    let pos_ratio = (pos + 1.0) / (total + 2.0);

    let new_pers = clonal_selection_update(*self.personality.read(), positive);
    *self.personality.write() = new_pers;

    let mut s = self.settings.write();
    s.recency_weight = s.recency_weight * 0.9 + (if pos_ratio > 0.6 { 0.25 } else { 0.15 }) * 0.1;
    s.importance_weight = s.importance_weight * 0.9 + (if pos_ratio > 0.6 { 0.35 } else { 0.25 }) * 0.1;
    s.similarity_weight = 1.0 - s.recency_weight - s.importance_weight;

    // Trust builds slowly; starts low
    let session_bonus = (*self.session_count.read() as f32 / 30.0).min(0.2);
    *self.trust_level.write() = (pos_ratio * 0.6 + session_bonus).clamp(0.15, 0.95);
    self.trigger_save();
}
```

```rust
pub fn retrieve(&mut self, query: &str, top_k: usize) -> Vec<String> {
    let current_time = Self::now_secs();
    let query_hash = text_hash(&query.to_lowercase().trim());
    // ... cache check ...

    // Similarity-based candidates
    let candidates = index.query(query_hash);
    let memories = self.memories.read();
    let mut scored: Vec<(f32, usize)> = candidates.into_iter()
        .map(|idx| (self.harmony_score(&memories[idx], similarity(query_hash, memories[idx].hash), current_time), idx))
        .collect();
    scored.sort_by(|a,b| b.0.partial_cmp(&a.0).unwrap());

    // Hybrid selection: 3 similarity + 2 recency + 1 importance
    let mut selected = Vec::new();
    let mut seen = HashSet::new();
    for (_, idx) in scored.iter().take(3) { if seen.insert(*idx) { selected.push(*idx); } }
    // Recency
    let mut by_recency: Vec<_> = (0..memories.len()).collect();
    by_recency.sort_by_key(|&i| -(memories[i].timestamp as i64));
    for idx in by_recency { if seen.insert(idx) && !selected.contains(&idx) { selected.push(idx); if selected.len() >= top_k { break; } } }
    // Importance
    let mut by_imp: Vec<_> = (0..memories.len()).collect();
    by_imp.sort_by_key(|&i| -(memories[i].importance as i32));
    for idx in by_imp { if seen.insert(idx) && !selected.contains(&idx) { selected.push(idx); if selected.len() >= top_k { break; } } }

    let results: Vec<String> = selected.iter().map(|&i| memories[i].text.clone()).collect();
    // ... stdp recording ...
    results
}
```

### STDP with Diminishing Returns

```rust
pub fn apply_stdp_batch(&mut self, action_time: f64) {
    let mut stdp = self.stdp_updater.write();
    let mut memories = self.memories.write();
    let mut importance_set = self.importance_set.write();
    stdp.apply_pending(action_time, |memory_id, delta| {
        if let Some(mem) = memories.iter_mut().find(|m| m.id == memory_id) {
            if !mem.permanent { importance_set.remove(&ImportanceEntry{importance:mem.importance, id:mem.id}); }
            // Diminishing returns
            let scale = 1.0 - (mem.importance as f32 / 65535.0);
            let effective_delta = delta * scale;
            let new_imp = (mem.importance as f32 + effective_delta * 65535.0).clamp(0.0, 65535.0) as u16;
            mem.importance = new_imp;
            if !mem.permanent { importance_set.insert(ImportanceEntry{importance:new_imp, id:mem.id}); }
        }
    });
    self.index_dirty.set(true);
}
```

### Pruning with ID-based Low-Importance List

```rust
fn prune_low_importance(&mut self) {
    let mut memories = self.memories.write();
    let mut importance_set = self.importance_set.write();
    while memories.len() > MAX_MEM {
        if let Some(entry) = importance_set.first().copied() {
            importance_set.remove(&entry);
            if let Some(pos) = memories.iter().position(|m| m.id == entry.id && !m.permanent) {
                memories.swap_remove(pos);
            }
        } else { break; }
    }
    // Rebuild low-importance ID list (not indices)
    let low: Vec<usize> = memories.iter()
        .filter(|m| m.importance < PRUNE_THRESHOLD && !m.permanent)
        .map(|m| m.id)
        .collect();
    *self.low_importance_ids.write() = low;
    self.index_dirty.set(true);
}
```

```rust
        ("sadness", vec!["I hear sadness in your voice.", "That feels heavy.", "I'm sorry you're going through this."]),
        ("excitement", vec!["I hear excitement!", "That's wonderful!", "Your energy is contagious!"]),

        ("confusion", vec!["I hear confusion.", "That's puzzling.", "Let's untangle this together."]),
    ];
    if let Some(variants) = templates.iter().find(|(e,_)| *e == emotion).map(|(_,v)| v) {

    } else {
        String::new()
    }
}
```
