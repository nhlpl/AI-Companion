```rust
use ahash::AHasher;
use std::hash::{Hash, Hasher};

pub fn normalize_text(text: &str) -> String {
    text.to_lowercase()
        .trim()
        .chars()
        .filter(|c| c.is_alphanumeric() || c.is_whitespace())
        .collect()
}

pub fn text_hash(text: &str) -> u64 {
    let normalized = normalize_text(text);
    let mut hasher = AHasher::default();
    normalized.hash(&mut hasher);
    hasher.finish()
}

pub fn similarity(h1: u64, h2: u64) -> f32 {
    let diff = (h1 ^ h2).count_ones();
    1.0 - (diff as f32 / 64.0)
}
```

```rust
use serde::{Deserialize, Serialize};
use crate::utils::constants::BLOOM_SIZE;

#[derive(Serialize, Deserialize, Clone)]
pub struct BloomFilter {
    bits: Vec<u64>,
    ring: Vec<u64>,
    pos: usize,
    capacity: usize,
}

impl BloomFilter {
    pub fn new(capacity: usize) -> Self {
        let words = (BLOOM_SIZE + 63) / 64;
        Self { bits: vec![0; words], ring: vec![0; capacity], pos: 0, capacity }
    }

    fn hash_pair(h: u64) -> (usize, usize) {
        let h1 = h.wrapping_mul(0x9e3779b97f4a7c15);
        let h2 = h.wrapping_mul(0xbf58476d1ce4e5b9);
        (h1 as usize % BLOOM_SIZE, h2 as usize % BLOOM_SIZE)
    }

    pub fn insert(&mut self, hash: u64) {
        let (i1, i2) = Self::hash_pair(hash);
        self.bits[i1 / 64] |= 1 << (i1 % 64);
        self.bits[i2 / 64] |= 1 << (i2 % 64);
        self.ring[self.pos] = hash;
        self.pos = (self.pos + 1) % self.capacity;
    }

    pub fn exact_duplicate(&self, hash: u64) -> bool {
        self.ring.contains(&hash)
    }
}
```

```rust
use serde::{Deserialize, Serialize};
use std::collections::HashSet;
use crate::domain::memory::Memory;

#[derive(Serialize, Deserialize, Clone)]
pub struct InvertedIndex {
    buckets: [Vec<Vec<usize>>; 8],
}

impl InvertedIndex {
    pub fn new() -> Self {
        const EMPTY: Vec<Vec<usize>> = Vec::new();
        let mut buckets = [EMPTY; 8];
        for i in 0..8 {
            buckets[i] = vec![Vec::new(); 256];
        }
        Self { buckets }
    }

    pub fn clear(&mut self) {
        for pos in 0..8 {
            for byte in 0..256 {
                self.buckets[pos][byte].clear();
            }
        }
    }

    pub fn insert(&mut self, hash: u64, idx: usize) {
        let bytes = hash.to_le_bytes();
        for (pos, &byte) in bytes.iter().enumerate() {
            self.buckets[pos][byte as usize].push(idx);
        }
    }

    pub fn build(&mut self, memories: &[Memory]) {
        self.clear();
        for (idx, mem) in memories.iter().enumerate() {
            self.insert(mem.hash, idx);
        }
    }

    pub fn query(&self, query_hash: u64) -> Vec<usize> {
        let bytes = query_hash.to_le_bytes();
        let mut candidates = Vec::new();
        let mut seen = HashSet::new();
        let mut pos_sizes: Vec<_> = (0..4)
            .map(|pos| (pos, self.buckets[pos][bytes[pos] as usize].len()))
            .collect();
        pos_sizes.sort_by_key(|&(_, sz)| sz);
        for (pos, _) in pos_sizes.iter().take(2) {
            for &idx in &self.buckets[*pos][bytes[*pos] as usize] {
                if seen.insert(idx) {
                    candidates.push(idx);
                }
            }
        }
        candidates
    }
}
```

```rust
// Power-law recency cache
pub struct RecencyCache {
    table: Vec<f32>,
}

impl RecencyCache {
    pub fn new() -> Self {
        let mut table = Vec::with_capacity(3651);
        for i in 0..=3650 {
            let days = i as f32 * 0.1;
            table.push((1.0 + FORGETTING_BETA * days).powf(-FORGETTING_ALPHA));
        }
        Self { table }
    }
    pub fn get(&self, days: f32) -> f32 {
        let idx = (days * 10.0).round() as usize;
        if idx >= self.table.len() {
            self.table.last().copied().unwrap_or(0.0)
        } else {
            self.table[idx]
        }
    }
}

// STDP batch updater
#[derive(Clone)]
pub struct StdpBatchEntry {
    pub memory_id: usize,
    pub retrieval_time: f64,
}

pub struct StdpUpdater {
    pending: VecDeque<StdpBatchEntry>,
    pot_lookup: Vec<f32>,
    dep_lookup: Vec<f32>,
}

impl StdpUpdater {
    pub fn new() -> Self {
        let steps = 1000;
        let max_dt = STDP_MAX_DT;
        let mut pot_lookup = Vec::with_capacity(steps);
        let mut dep_lookup = Vec::with_capacity(steps);
        for i in 0..steps {
            let dt = (i as f32 / steps as f32) * max_dt;
            pot_lookup.push(STDP_A_PLUS * (-dt / STDP_TAU_PLUS).exp());
            dep_lookup.push(-STDP_A_MINUS * (dt / STDP_TAU_MINUS).exp());
        }
        Self { pending: VecDeque::new(), pot_lookup, dep_lookup }
    }

    pub fn record(&mut self, memory_id: usize, retrieval_time: f64) {
        self.pending.push_back(StdpBatchEntry { memory_id, retrieval_time });
        if self.pending.len() > 200 { self.pending.pop_front(); }
    }

    pub fn apply_pending<F>(&mut self, action_time: f64, mut update_fn: F)
    where F: FnMut(usize, f32)
    {
        let steps = self.pot_lookup.len();
        let max_dt = STDP_MAX_DT;
        while let Some(entry) = self.pending.pop_front() {
            let dt = action_time - entry.retrieval_time;
            if dt.abs() > max_dt as f64 { continue; }
            let idx = ((dt.abs() as f32 / max_dt) * (steps - 1) as f32).round() as usize;
            let delta = if dt > 0.0 { self.pot_lookup[idx] } else { self.dep_lookup[idx] };
            update_fn(entry.memory_id, delta);
        }
    }
}

        let avg_word_len = if words.is_empty() { 0.0 } else {

            Self::Analytical
        } else if has_exclamation || personality > 10 {
            Self::Playful
        } else {
            Self::Formal
        }
    }
}

pub fn clonal_selection_update(current: u8, positive: bool) -> u8 {
    let cur = current as f32 / 15.0;
    let mut rng = rand::thread_rng();
    let clones: Vec<f32> = (0..5).map(|_| {
        let noise = rng.gen_range(-0.1..0.1);
        (cur + noise).clamp(0.0, 1.0)

    let target = if positive { 1.0 } else { 0.0 };
    let best = clones.into_iter().min_by(|a,b| (a-target).abs().partial_cmp(&(b-target).abs()).unwrap()).unwrap();

}
```

```rust
use std::collections::{BTreeSet, HashSet};
use std::num::NonZeroUsize;
use lru::LruCache;
use rand::Rng;

use crate::utils::*;
use crate::domain::memory::Memory;
use crate::domain::bio_features::{RecencyCache, StdpUpdater};

#[derive(Debug, Clone, Copy, PartialEq, Eq)]
pub struct ImportanceEntry {
    pub importance: u16,
    pub id: usize,
}

impl Ord for ImportanceEntry {
    fn cmp(&self, o: &Self) -> std::cmp::Ordering {
        self.importance.cmp(&o.importance).then(self.id.cmp(&o.id))
    }
}

impl PartialOrd for ImportanceEntry {
    fn partial_cmp(&self, o: &Self) -> Option<std::cmp::Ordering> {
        Some(self.cmp(o))
    }
}

pub struct MemoryEngine {
    pub memories: Vec<Memory>,
    pub inverted_index: InvertedIndex,
    pub bloom: BloomFilter,
    pub importance_set: BTreeSet<ImportanceEntry>,
    pub low_importance_ids: Vec<usize>,
    pub cache: LruCache<u64, Vec<usize>>,
    pub index_dirty: bool,
    pub next_id: usize,
    pub recency_cache: RecencyCache,
    pub stdp_updater: StdpUpdater,
}

impl MemoryEngine {
    pub fn new() -> Self {
        Self {
            memories: Vec::new(),
            inverted_index: InvertedIndex::new(),
            bloom: BloomFilter::new(128),
            importance_set: BTreeSet::new(),
            low_importance_ids: Vec::new(),
            cache: LruCache::new(NonZeroUsize::new(CACHE_SIZE).unwrap()),
            index_dirty: true,
            next_id: 0,
            recency_cache: RecencyCache::new(),
            stdp_updater: StdpUpdater::new(),
        }
    }

    pub fn load(memories: Vec<Memory>, next_id: usize) -> Self {
        let mut engine = Self::new();
        for mem in &memories {
            engine.bloom.insert(mem.hash);
            if !mem.permanent {
                engine.importance_set.insert(ImportanceEntry { importance: mem.importance, id: mem.id });
            }
            if mem.importance < PRUNE_THRESHOLD && !mem.permanent {
                engine.low_importance_ids.push(mem.id);
            }
        }
        engine.memories = memories;
        engine.next_id = next_id;
        engine.inverted_index.build(&engine.memories);
        engine.index_dirty = false;
        engine
    }

    pub fn harmony_score(&self, mem: &Memory, sim: f32, current_time: f64, weights: (f32, f32, f32)) -> f32 {
        let days = ((current_time - mem.timestamp) / 86400.0) as f32;
        let recency = self.recency_cache.get(days);
        let importance = mem.importance as f32 / 65535.0;
        weights.0 * sim + weights.1 * importance + weights.2 * recency
    }

    pub fn add(&mut self, text: String, importance: u16, permanent: bool, now: f64) -> usize {
        let hash = text_hash(&text);
        if self.bloom.exact_duplicate(hash) {
            if let Some(pos) = self.memories.iter().position(|m| m.hash == hash && m.text == text) {
                let mem = &mut self.memories[pos];
                if importance > mem.importance {
                    if !mem.permanent {
                        self.importance_set.remove(&ImportanceEntry { importance: mem.importance, id: mem.id });
                    }
                    mem.importance = importance;
                    mem.timestamp = now;
                    mem.permanent = permanent;
                    if !permanent {
                        self.importance_set.insert(ImportanceEntry { importance, id: mem.id });
                    }
                    self.index_dirty = true;
                }
                return mem.id;
            }
        }
        let id = self.next_id;
        self.next_id += 1;
        let mem = Memory::new(id, text, importance, permanent, now);
        self.bloom.insert(hash);
        self.memories.push(mem.clone());
        self.inverted_index.insert(hash, self.memories.len() - 1);
        if !permanent {
            self.importance_set.insert(ImportanceEntry { importance, id });
        }
        if importance < PRUNE_THRESHOLD && !permanent {
            self.low_importance_ids.push(id);
        }
        self.index_dirty = true;
        if self.memories.len() > MAX_MEM {
            self.prune_low_importance();
        }
        id
    }

    fn prune_low_importance(&mut self) {
        while self.memories.len() > MAX_MEM {
            if let Some(entry) = self.importance_set.first().copied() {
                self.importance_set.remove(&entry);
                if let Some(pos) = self.memories.iter().position(|m| m.id == entry.id && !m.permanent) {
                    self.memories.swap_remove(pos);
                }
            } else {
                break;
            }
        }
        self.low_importance_ids = self.memories.iter()
            .filter(|m| m.importance < PRUNE_THRESHOLD && !m.permanent)
            .map(|m| m.id)
            .collect();
        self.index_dirty = true;
    }

    pub fn retrieve(&mut self, query: &str, top_k: usize, current_time: f64, weights: (f32, f32, f32), harmony_mode: bool) -> Vec<String> {
        let normalized = normalize_text(query);
        let query_hash = text_hash(&normalized);

        if let Some(cached) = self.cache.get(&query_hash) {
            let results: Vec<String> = cached.iter()
                .filter_map(|&id| self.memories.iter().find(|m| m.id == id))
                .take(top_k)
                .map(|m| m.text.clone())
                .collect();
            for id in cached.iter().take(top_k) {
                self.stdp_updater.record(*id, current_time);
            }
            return results;
        }

        if self.index_dirty {
            self.inverted_index.build(&self.memories);
            self.index_dirty = false;
        }

        let candidates = self.inverted_index.query(query_hash);
        let mut scored: Vec<(f32, usize)> = candidates.into_iter()
            .map(|idx| {
                let mem = &self.memories[idx];
                let sim = similarity(query_hash, mem.hash);
                let score = if harmony_mode {
                    self.harmony_score(mem, sim, current_time, weights)
                } else {
                    sim
                };
                (score, idx)
            })
            .collect();
        scored.sort_by(|a, b| b.0.partial_cmp(&a.0).unwrap());

        let mut selected = Vec::new();
        let mut seen = HashSet::new();
        for (_, idx) in scored.iter().take(3) {
            if seen.insert(*idx) { selected.push(*idx); }
        }
        let mut by_recency: Vec<_> = (0..self.memories.len()).collect();
        by_recency.sort_by_key(|&i| -(self.memories[i].timestamp as i64));
        for idx in by_recency {
            if seen.insert(idx) && !selected.contains(&idx) {
                selected.push(idx);
                if selected.len() >= top_k { break; }
            }
        }
        let mut by_imp: Vec<_> = (0..self.memories.len()).collect();
        by_imp.sort_by_key(|&i| -(self.memories[i].importance as i32));
        for idx in by_imp {
            if seen.insert(idx) && !selected.contains(&idx) {
                selected.push(idx);
                if selected.len() >= top_k { break; }
            }
        }

        if harmony_mode && rand::thread_rng().gen_bool(SERENDIPITY_CHANCE) {
            if !self.low_importance_ids.is_empty() {
                let rand_id = self.low_importance_ids[rand::thread_rng().gen_range(0..self.low_importance_ids.len())];
                if let Some(idx) = self.memories.iter().position(|m| m.id == rand_id) {
                    if !selected.contains(&idx) && selected.len() > 1 {
                        selected[1] = idx;
                    }
                }
            }
        }

        let results: Vec<String> = selected.iter().map(|&i| self.memories[i].text.clone()).collect();
        for &id in &selected {
            self.stdp_updater.record(id, current_time);
        }
        self.cache.put(query_hash, selected.clone());
        results
    }

    pub fn apply_stdp_batch(&mut self, action_time: f64) {
        self.stdp_updater.apply_pending(action_time, |memory_id, delta| {
            if let Some(mem) = self.memories.iter_mut().find(|m| m.id == memory_id) {
                if !mem.permanent {
                    self.importance_set.remove(&ImportanceEntry { importance: mem.importance, id: mem.id });
                }
                let scale = 1.0 - (mem.importance as f32 / 65535.0);
                let effective_delta = delta * scale;
                let new_imp = (mem.importance as f32 + effective_delta * 65535.0).clamp(0.0, 65535.0) as u16;
                mem.importance = new_imp;
                if !mem.permanent {
                    self.importance_set.insert(ImportanceEntry { importance: new_imp, id: mem.id });
                }
            }
        });
        self.index_dirty = true;
    }

    pub fn get_random_memory_fragment(&self) -> Option<String> {
        if self.memories.is_empty() { return None; }
        let idx = rand::thread_rng().gen_range(0..self.memories.len());
        let text = &self.memories[idx].text;
        Some(text.split_whitespace().take(3).collect::<Vec<_>>().join(" "))
    }
}
```

```rust
}

impl SettingsEngine {
    pub fn new() -> Self {
        Self {
            personality: 7,
            settings: Settings::default(),
            trust_level: 0.15,
            session_count: 1,
            feedback_history: VecDeque::new(),
        }
    }

    pub fn record_feedback(&mut self, positive: bool, now: f64) {
        self.feedback_history.retain(|(ts, _)| now - ts < 7.0 * 86400.0);
        self.feedback_history.push_back((now, positive));
        if self.feedback_history.len() > 50 {
            self.feedback_history.pop_front();
        }

        let pos = self.feedback_history.iter().filter(|(_, p)| *p).count() as f32;
        let total = self.feedback_history.len() as f32;
        let pos_ratio = (pos + 1.0) / (total + 2.0);

        self.personality = clonal_selection_update(self.personality, positive);

        self.settings.recency_weight = self.settings.recency_weight * 0.9 + (if pos_ratio > 0.6 { 0.25 } else { 0.15 }) * 0.1;
        self.settings.importance_weight = self.settings.importance_weight * 0.9 + (if pos_ratio > 0.6 { 0.35 } else { 0.25 }) * 0.1;
        self.settings.similarity_weight = 1.0 - self.settings.recency_weight - self.settings.importance_weight;

        let session_bonus = (self.session_count as f32 / 30.0).min(0.2);
        self.trust_level = (pos_ratio * 0.7 + session_bonus).clamp(0.15, 0.98);
    }

    pub fn harmony_weights(&self) -> (f32, f32, f32) {
        (self.settings.similarity_weight, self.settings.importance_weight, self.settings.recency_weight)
    }
}
```

```rust
        let persisted = PersistedState {
            memories: self.memory_engine.read().memories.clone(),
            anti_memories: self.anti_memories.read().clone(),
            personality: self.settings_engine.read().personality,
            settings: self.settings_engine.read().settings.clone(),

            trust_level: self.settings_engine.read().trust_level,

    pub fn update_archetype(&mut self, user_message: &str) {
        let msgs = self.chat_messages.read();
        let user_msgs: Vec<_> = msgs.iter().filter(|m| m.role=="user").map(|m| m.content.as_str()).collect();
        let q_ratio = user_msgs.iter().filter(|m| m.contains('?')).count() as f32 / user_msgs.len().max(1) as f32;
        let sentiment = self.analyze_sentiment(&user_msgs);
        let arch = if q_ratio > 0.3 { Archetype::Mentor } else if sentiment > 0.5 { Archetype::Companion } else { Archetype::Trickster };
        *self.current_archetype.write() = arch;
    }

    fn analyze_sentiment(&self, msgs: &[&str]) -> f32 {
        let pos_w = ["good","great","happy","love","excited"];
        let neg_w = ["bad","sad","angry","hate","terrible"];
        let mut pos=0; let mut neg=0;
        for msg in msgs {
            for w in msg.split_whitespace() {
                let l = w.to_lowercase();
                if pos_w.iter().any(|&pw| l.contains(pw)) { pos+=1; }
                if neg_w.iter().any(|&nw| l.contains(nw)) { neg+=1; }
            }
        }
        if pos+neg==0 { 0.5 } else { pos as f32 / (pos+neg) as f32 }
    }

    fn detect_emotion(&self, text: &str) -> String {
        let l = text.to_lowercase();
        if l.contains("angry")||l.contains("frustrated") {"frustration"}
        else if l.contains("sad")||l.contains("upset") {"sadness"}
        else if l.contains("happy")||l.contains("excited") {"excitement"}
        else if l.contains("anxious")||l.contains("worried") {"anxiety"}
        else if l.contains("confused") {"confusion"}
        else {""}.to_string()
    }

    fn get_emotion_validation(&self, emotion: &str) -> String {
        let templates = [
            ("frustration", vec!["I hear frustration in your words.", "That sounds really frustrating.", "I can sense the tension."]),
            ("sadness", vec!["I hear sadness in your voice.", "That feels heavy.", "I'm sorry you're going through this."]),
            ("excitement", vec!["I hear excitement!", "That's wonderful!", "Your energy is contagious!"]),
            ("anxiety", vec!["I hear anxiety creeping in.", "That sounds nerve-wracking.", "Take a breath with me."]),
            ("confusion", vec!["I hear confusion.", "That's puzzling.", "Let's untangle this together."]),
        ];
        if let Some(variants) = templates.iter().find(|(e,_)| *e == emotion).map(|(_,v)| v) {
            let mut usage = self.validation_usage.write();
            let count = usage.entry(emotion.to_string()).or_insert(0);
            let idx = (*count as usize) % variants.len();
            *count += 1;
            variants[idx].to_string()
        } else { String::new() }
    }

    fn extract_common_theme(&self, responses: &[String]) -> String {
        let stopwords: std::collections::HashSet<&str> = ["this","that","with","from","have","were","they","what","when","then","there","their","about","which","would","could","should","been","being","doing","just","like","very","really","some","more","connection"].iter().copied().collect();
        let mut counts = std::collections::HashMap::new();
        for r in responses {
            for w in r.split_whitespace() {
                let wc = w.to_lowercase().chars().filter(|c| c.is_alphabetic()).collect::<String>();
                if wc.len() > 4 && !stopwords.contains(wc.as_str()) {
                    *counts.entry(wc).or_insert(0) += 1;
                }
            }
        }
        let mut sorted: Vec<_> = counts.into_iter().collect();
        sorted.sort_by_key(|(_,c)| -(*c as i32));
        sorted.first().map(|(w,_)| w.clone()).unwrap_or_else(|| "this".to_string())
    }

    pub fn generate_response(&self, user_input: &str, base_response: String) -> String {
        let mut parts = Vec::new();
        let emotion = self.detect_emotion(user_input);
        if !emotion.is_empty() {
            parts.push(self.get_emotion_validation(&emotion));
        }
        let negative_emotions = ["frustration", "sadness", "anxiety"];
        if rand::thread_rng().gen_bool(0.25) && !negative_emotions.contains(&emotion.as_str()) {
            parts.push("Before I answer, what made this come to mind? ".to_string());
        }
        let arch = *self.current_archetype.read();
        parts.push(match arch { Archetype::Mentor=>"Let me share a perspective... ", Archetype::Companion=>"I'm glad you shared that. ", Archetype::Trickster=>"Hmm, interesting! " }.to_string());
        parts.push(base_response);
        if rand::thread_rng().gen_bool(0.15) {
            let recent = self.get_last_responses();
            if !recent.is_empty() {
                let theme = self.extract_common_theme(&recent);
                parts.push(format!(" I realize I've been circling around '{}' lately—maybe that's where we're both focused.", theme));
            }
        }
        if rand::thread_rng().gen_bool(0.2) && user_input.contains('?') {
            parts.push(" I have a gut feeling about this, though I can't fully explain why.".to_string());
        }
        let trust = self.settings_engine.read().trust_level;
        if trust > 0.6 && rand::thread_rng().gen_bool(0.15) {
            parts.insert(0, "...".to_string());
        }
        if trust > 0.7 && rand::thread_rng().gen_bool(0.1) {
            parts.push(" Just between us.".to_string());
        }
        if trust > 0.5 && rand::thread_rng().gen_bool(0.12) {
            if let Some(frag) = self.memory_engine.read().get_random_memory_fragment() {
                parts.insert(0, format!("I was thinking about when you mentioned '{}'... ", frag));
            }
        }
        if trust > 0.4 && rand::thread_rng().gen_bool(0.08) {
            let hour = Local::now().hour();
            if hour >= 20 || hour < 5 {
                parts.insert(0, "I imagine you're winding down. ".to_string());
            } else if hour < 12 {
                parts.insert(0, "I imagine you're just starting your day. ".to_string());
            }
        }
        if rand::thread_rng().gen_bool(0.1) {
            parts.insert(0, format!("{} ", vulnerability_cue()));
        }
        let mut response = parts.join("");
        if let Some(narrative) = get_canned_micronarrative() {
            response.push_str(&format!("\n\n{}", narrative));
        }
        if should_suggest_social_bridge(self.settings_engine.read().session_count, trust) {
            response.push_str(&format!("\n\n{}", generate_social_bridge_nudge(None)));
        }
        response
    }
}
```

```rust
                        s.personality = (s.personality as i16 + 1).clamp(0, 15) as u8;
                        state.trigger_save();
                    }, "+" }
                    button { class: "px-4 py-2 bg-blue-600 rounded", onclick: move |_| {
                        let mut s = settings.write();
                        s.personality = (s.personality as i16 - 1).clamp(0, 15) as u8;
                        state.trigger_save();
                    }, "-" }
                }
            }
            hr { class: "border-gray-700" }
            div { class: "space-y-3",
                h3 { class: "text-lg" "Harmony Weights" }
                p { "Similarity: {settings.read().settings.similarity_weight:.2}" }
                p { "Importance: {settings.read().settings.importance_weight:.2}" }
                p { "Recency: {settings.read().settings.recency_weight:.2}" }
                p { class: "text-sm text-gray-400" "Trust: {settings.read().trust_level:.2}" }
            }
            hr { class: "border-gray-700" }
            div { class: "space-y-3",
                h3 { class: "text-lg" "Anti‑Memory" }
                input { class: "w-full px-4 py-2 bg-gray-700 rounded-full text-white", placeholder: "Incorrect phrase", value: "{anti_incorrect}", oninput: move |e| anti_incorrect.set(e.value()) }
                input { class: "w-full px-4 py-2 bg-gray-700 rounded-full text-white", placeholder: "Correct phrase", value: "{anti_correct}", oninput: move |e| anti_correct.set(e.value()) }
                button { class: "px-4 py-2 bg-blue-600 rounded-full", onclick: move |_| {
                    let inc = anti_incorrect.read().clone(); let cor = anti_correct.read().clone();
                    if !inc.is_empty() && !cor.is_empty() { state.add_anti_memory(inc, cor); *anti_incorrect.write()=String::new(); *anti_correct.write()=String::new(); }
                }, "Add Correction" }
            }
        }
    }
```
