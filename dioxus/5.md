pub enum Archetype {

    /// Generate a response using living behaviors unconditionally.
    pub fn generate_response(&self, user_input: &str, base_response: String) -> String {
        let mut parts = Vec::new();

        // 1. Emotional validation
        let emotion = self.detect_emotion(user_input);
        if !emotion.is_empty() {
            parts.push(format!("I hear {} in your words. ", emotion));
        }

        // 2. Recursive curiosity (25% chance)
        if rand::thread_rng().gen_bool(0.25) {
            parts.push("Before I answer, what made this come to mind? ".to_string());
        }

        // 3. Archetype flavor
        let archetype = *self.current_archetype.read();
        let flavor = match archetype {
            Archetype::Mentor => "Let me share a perspective... ",
            Archetype::Companion => "I'm glad you shared that. ",
            Archetype::Trickster => "Hmm, interesting! ",
        };
        parts.push(flavor.to_string());

        // 4. The actual response
        parts.push(base_response);

        // 5. Strange loop (15% chance)
        if rand::thread_rng().gen_bool(0.15) {
            let recent = self.get_last_responses();
            if !recent.is_empty() {
                let theme = self.extract_common_theme(&recent);
                parts.push(format!(
                    " I realize I've been circling around themes like '{}' lately—maybe that's where we're both focused.",
                    theme
                ));
            }
        }

        // 6. Somatic marker (20% chance on questions)
        if rand::thread_rng().gen_bool(0.2) && user_input.contains('?') {
            parts.push(" I have a gut feeling about this, though I can't fully explain why.".to_string());
        }

        parts.join("")
    }

    // Record response for strange loop
    pub fn record_response(&mut self, response: &str) {
        let mut buffer = self.last_response_buffer.write();
        buffer.push_back(response.to_string());
        if buffer.len() > 5 {
            buffer.pop_front();
        }
    }

    pub fn get_last_responses(&self) -> Vec<String> {
        self.last_response_buffer.read().iter().cloned().collect()
    }

    // Update archetype based on user interaction patterns
    pub fn update_archetype(&mut self, user_message: &str) {
        let messages = self.chat_messages.read();
        let user_msgs: Vec<_> = messages
            .iter()
            .filter(|m| m.role == "user")
            .map(|m| m.content.as_str())
            .collect();

        let question_ratio = user_msgs
            .iter()
            .filter(|m| m.contains('?'))
            .count() as f32 / user_msgs.len().max(1) as f32;

        let sentiment = self.analyze_sentiment(&user_msgs);

        let archetype = if question_ratio > 0.3 {
            Archetype::Mentor
        } else if sentiment > 0.5 {
            Archetype::Companion
        } else {
            Archetype::Trickster
        };

        *self.current_archetype.write() = archetype;
    }

    // Simple sentiment analysis (positive/negative word counts)
    fn analyze_sentiment(&self, messages: &[&str]) -> f32 {
        let positive_words = ["good", "great", "happy", "love", "excited", "wonderful"];
        let negative_words = ["bad", "sad", "angry", "hate", "terrible", "upset"];
        let mut pos = 0;
        let mut neg = 0;
        for msg in messages {
            for word in msg.split_whitespace() {
                let low = word.to_lowercase();
                if positive_words.iter().any(|&pw| low.contains(pw)) {
                    pos += 1;
                }
                if negative_words.iter().any(|&nw| low.contains(nw)) {
                    neg += 1;
                }
            }
        }
        if pos + neg == 0 {
            0.5
        } else {
            pos as f32 / (pos + neg) as f32
        }
    }

    // Detect primary emotion in text
    fn detect_emotion(&self, text: &str) -> String {
        let low = text.to_lowercase();
        if low.contains("angry") || low.contains("frustrated") {
            "frustration"
        } else if low.contains("sad") || low.contains("upset") {
            "sadness"
        } else if low.contains("happy") || low.contains("excited") {
            "excitement"
        } else if low.contains("anxious") || low.contains("worried") {
            "anxiety"
        } else if low.contains("confused") {
            "confusion"
        } else {
            ""
        }
        .to_string()
    }

    // Extract a common theme from recent responses
    fn extract_common_theme(&self, responses: &[String]) -> String {
        let mut word_counts = std::collections::HashMap::new();
        for r in responses {
            for word in r.split_whitespace() {
                let w = word.to_lowercase();
                if w.len() > 4 {
                    *word_counts.entry(w).or_insert(0) += 1;
                }
            }
        }
        let mut sorted: Vec<_> = word_counts.into_iter().collect();
        sorted.sort_by_key(|(_, c)| -(*c as i32));
        sorted
            .first()
            .map(|(w, _)| w.clone())
            .unwrap_or_else(|| "connection".to_string())
    }
