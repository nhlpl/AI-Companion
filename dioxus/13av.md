use std::collections::{BTreeSet, HashSet};
use std::num::NonZeroUsize;
use lru::LruCache;
use rand::Rng;
use crate::utils::*;
use crate::domain::memory::Memory;
use crate::domain::bio_features::{RecencyCache, StdpUpdater};

#[derive(Debug, Clone, Copy, PartialEq, Eq)]
pub struct ImportanceEntry { pub importance: u16, pub id: usize }
impl Ord for ImportanceEntry { fn cmp(&self, o: &Self) -> std::cmp::Ordering { self.importance.cmp(&o.importance).then(self.id.cmp(&o.id)) } }
impl PartialOrd for ImportanceEntry { fn partial_cmp(&self, o: &Self) -> Option<std::cmp::Ordering> { Some(self.cmp(o)) } }

pub struct MemoryEngine {
    pub memories: Vec<Memory>,
    pub inverted_index: InvertedIndex,
    pub bloom: BloomFilter,
    pub importance_set: BTreeSet<ImportanceEntry>,
    pub low_importance_ids: Vec<usize>,
    pub cache: LruCache<u64, Vec<usize>>,
    pub index_dirty: bool,
    pub next_id: usize,
    pub recency_cache: RecencyCache,
    pub stdp_updater: StdpUpdater,
}

impl MemoryEngine {
    pub fn new() -> Self {
        Self { memories: Vec::new(), inverted_index: InvertedIndex::new(), bloom: BloomFilter::new(128), importance_set: BTreeSet::new(), low_importance_ids: Vec::new(), cache: LruCache::new(NonZeroUsize::new(CACHE_SIZE).unwrap()), index_dirty: true, next_id: 0, recency_cache: RecencyCache::new(), stdp_updater: StdpUpdater::new() }
    }
    pub fn load(mems: Vec<Memory>, next_id: usize) -> Self {
        let mut e = Self::new();
        for m in &mems { e.bloom.insert(m.hash); if !m.permanent { e.importance_set.insert(ImportanceEntry{importance:m.importance, id:m.id}); } if m.importance<PRUNE_THRESHOLD && !m.permanent { e.low_importance_ids.push(m.id); } }
        e.memories = mems; e.next_id = next_id; e.inverted_index.build(&e.memories); e.index_dirty = false; e
    }
    pub fn harmony_score(&self, mem: &Memory, sim: f32, cur: f64, w: (f32,f32,f32)) -> f32 {
        let days = ((cur - mem.timestamp)/86400.0) as f32; w.0*sim + w.1*(mem.importance as f32/65535.0) + w.2*self.recency_cache.get(days)
    }
    pub fn add(&mut self, text: String, imp: u16, perm: bool, now: f64) -> usize {
        let hash = text_hash(&text);
        if self.bloom.exact_duplicate(hash) {
            if let Some(p) = self.memories.iter().position(|m| m.hash==hash && m.text==text) {
                let m = &mut self.memories[p];
                if imp > m.importance { if !m.permanent { self.importance_set.remove(&ImportanceEntry{importance:m.importance, id:m.id}); } m.importance=imp; m.timestamp=now; m.permanent=perm; if !perm { self.importance_set.insert(ImportanceEntry{importance:imp, id:m.id}); } self.index_dirty=true; }
                return m.id;
            }
        }
        let id = self.next_id; self.next_id += 1;
        let m = Memory::new(id, text, imp, perm, now);
        self.bloom.insert(hash); self.memories.push(m.clone()); self.inverted_index.insert(hash, self.memories.len()-1);
        if !perm { self.importance_set.insert(ImportanceEntry{importance:imp, id}); }
        if imp<PRUNE_THRESHOLD && !perm { self.low_importance_ids.push(id); }
        self.index_dirty = true;
        if self.memories.len() > MAX_MEM { self.prune(); }
        id
    }
    fn prune(&mut self) {
        while self.memories.len() > MAX_MEM { if let Some(e) = self.importance_set.first().copied() { self.importance_set.remove(&e); if let Some(p) = self.memories.iter().position(|m| m.id==e.id && !m.permanent) { self.memories.swap_remove(p); } } else { break; } }
        self.low_importance_ids = self.memories.iter().filter(|m| m.importance<PRUNE_THRESHOLD && !m.permanent).map(|m| m.id).collect(); self.index_dirty = true;
    }
    pub fn retrieve(&mut self, q: &str, top_k: usize, now: f64, w: (f32,f32,f32), harmony: bool) -> Vec<String> {
        let norm = normalize_text(q); let h = text_hash(&norm);
        if let Some(c) = self.cache.get(&h) { let r: Vec<_> = c.iter().filter_map(|&id| self.memories.iter().find(|m| m.id==id)).take(top_k).map(|m| m.text.clone()).collect(); for id in c.iter().take(top_k) { self.stdp_updater.record(*id, now); } return r; }
        if self.index_dirty { self.inverted_index.build(&self.memories); self.index_dirty=false; }
        let cand = self.inverted_index.query(h);
        let mut scored: Vec<(f32,usize)> = cand.into_iter().map(|idx| { let m=&self.memories[idx]; let sim=similarity(h, m.hash); (if harmony {self.harmony_score(m,sim,now,w)} else {sim}, idx) }).collect();
        scored.sort_by(|a,b| b.0.partial_cmp(&a.0).unwrap());
        let mut sel=Vec::new(); let mut seen=HashSet::new();
        for (_,idx) in scored.iter().take(3) { if seen.insert(*idx) { sel.push(*idx); } }
        let mut rec: Vec<_> = (0..self.memories.len()).collect(); rec.sort_by_key(|&i| -(self.memories[i].timestamp as i64));
        for idx in rec { if seen.insert(idx) && !sel.contains(&idx) { sel.push(idx); if sel.len()>=top_k { break; } } }
        let mut imp: Vec<_> = (0..self.memories.len()).collect(); imp.sort_by_key(|&i| -(self.memories[i].importance as i32));
        for idx in imp { if seen.insert(idx) && !sel.contains(&idx) { sel.push(idx); if sel.len()>=top_k { break; } } }
        if harmony && rand::thread_rng().gen_bool(SERENDIPITY_CHANCE) && !self.low_importance_ids.is_empty() {
            let rid = self.low_importance_ids[rand::thread_rng().gen_range(0..self.low_importance_ids.len())];
            if let Some(idx) = self.memories.iter().position(|m| m.id==rid) { if !sel.contains(&idx) && sel.len()>1 { sel[1]=idx; } }
        }
        let res: Vec<_> = sel.iter().map(|&i| self.memories[i].text.clone()).collect();
        for &id in &sel { self.stdp_updater.record(id, now); }
        self.cache.put(h, sel); res
    }
    pub fn apply_stdp_batch(&mut self, action: f64) {
        self.stdp_updater.apply_pending(action, |id, delta| {
            if let Some(m) = self.memories.iter_mut().find(|m| m.id==id) {
                if !m.permanent { self.importance_set.remove(&ImportanceEntry{importance:m.importance, id:m.id}); }
                let scale = 1.0 - (m.importance as f32/65535.0); let new = (m.importance as f32 + delta*scale*65535.0).clamp(0.0,65535.0) as u16;
                m.importance = new; if !m.permanent { self.importance_set.insert(ImportanceEntry{importance:new, id}); }
            }
        }); self.index_dirty = true;
    }
    pub fn get_random_fragment(&self) -> Option<String> { if self.memories.is_empty() { None } else { let i = rand::thread_rng().gen_range(0..self.memories.len()); Some(self.memories[i].text.split_whitespace().take(3).collect::<Vec<_>>().join(" ")) } }
}

        self.settings.importance_weight = self.settings.importance_weight*0.9 + (if ratio>0.6 {0.35} else {0.25})*0.1;

use serde::{Deserialize, Serialize};
use crate::utils::constants::BLOOM_SIZE;

#[derive(Serialize, Deserialize, Clone)]
pub struct BloomFilter {
    bits: Vec<u64>,
    ring: Vec<u64>,
    pos: usize,
    capacity: usize,
}

impl BloomFilter {
    pub fn new(capacity: usize) -> Self {
        let words = (BLOOM_SIZE + 63) / 64;
        Self { bits: vec![0; words], ring: vec![0; capacity], pos: 0, capacity }
    }

    fn hash_pair(h: u64) -> (usize, usize) {
        let h1 = h.wrapping_mul(0x9e3779b97f4a7c15);
        let h2 = h.wrapping_mul(0xbf58476d1ce4e5b9);
        (h1 as usize % BLOOM_SIZE, h2 as usize % BLOOM_SIZE)
    }

    pub fn insert(&mut self, hash: u64) {
        let (i1, i2) = Self::hash_pair(hash);
        self.bits[i1 / 64] |= 1 << (i1 % 64);
        self.bits[i2 / 64] |= 1 << (i2 % 64);
        self.ring[self.pos] = hash;
        self.pos = (self.pos + 1) % self.capacity;
    }

    pub fn exact_duplicate(&self, hash: u64) -> bool {
        self.ring.contains(&hash)
    }
}

use serde::{Deserialize, Serialize};
use std::collections::HashSet;
use crate::domain::memory::Memory;

#[derive(Serialize, Deserialize, Clone)]
pub struct InvertedIndex {
    buckets: [Vec<Vec<usize>>; 8],
}

impl InvertedIndex {
    pub fn new() -> Self {
        const EMPTY: Vec<Vec<usize>> = Vec::new();
        let mut buckets = [EMPTY; 8];
        for i in 0..8 {
            buckets[i] = vec![Vec::new(); 256];
        }
        Self { buckets }
    }

    pub fn clear(&mut self) {
        for pos in 0..8 {
            for byte in 0..256 {
                self.buckets[pos][byte].clear();
            }
        }
    }

    pub fn insert(&mut self, hash: u64, idx: usize) {
        let bytes = hash.to_le_bytes();
        for (pos, &byte) in bytes.iter().enumerate() {
            self.buckets[pos][byte as usize].push(idx);
        }
    }

    pub fn build(&mut self, memories: &[Memory]) {
        self.clear();
        for (idx, mem) in memories.iter().enumerate() {
            self.insert(mem.hash, idx);
        }
    }

    pub fn query(&self, query_hash: u64) -> Vec<usize> {
        let bytes = query_hash.to_le_bytes();
        let mut candidates = Vec::new();
        let mut seen = HashSet::new();
        let mut pos_sizes: Vec<_> = (0..4)
            .map(|pos| (pos, self.buckets[pos][bytes[pos] as usize].len()))
            .collect();
        pos_sizes.sort_by_key(|&(_, sz)| sz);
        for (pos, _) in pos_sizes.iter().take(2) {
            for &idx in &self.buckets[*pos][bytes[*pos] as usize] {
                if seen.insert(idx) {
                    candidates.push(idx);
                }
            }
        }
        candidates
    }
}
