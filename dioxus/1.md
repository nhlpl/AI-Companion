use dioxus_router::prelude::*;
use dioxus_sdk::storage::*;

use std::collections::{BinaryHeap, HashMap, HashSet};
use std::cmp::Reverse;
use std::sync::Arc;
use tokio::sync::RwLock;

use ahash::AHasher;
use std::hash::{Hash, Hasher};
use lru::LruCache;
use std::num::NonZeroUsize;
use chrono::Utc;

// ------------------------------
// Optimized constants
// ------------------------------
const MAX_MEM: usize = 25000;
const PRUNE_THRESHOLD: u16 = 655;
const EARLY_STOP_THRESHOLD: f32 = 0.3;
const CACHE_SIZE: usize = 100;
const BLOOM_SIZE: usize = 1024;
const SERENDIPITY_CHANCE: f64 = 0.05;
const ANTI_DECAY_RATE: f32 = 0.95;
const ANTI_STRENGTHEN_RATE: f32 = 1.1;
const RECENCY_LAMBDA: f64 = 0.1;

// ------------------------------
// Fast 64-bit text hash
// ------------------------------
fn text_hash(text: &str) -> u64 {
    let mut hasher = AHasher::default();
    text.hash(&mut hasher);
    hasher.finish()
}

fn similarity(h1: u64, h2: u64) -> f32 {
    let diff = (h1 ^ h2).count_ones();
    1.0 - (diff as f32 / 64.0)
}

// ------------------------------
// Bloom filter for deduplication
// ------------------------------
#[derive(Serialize, Deserialize, Clone)]
struct BloomFilter {
    bits: Vec<u64>,
    ring: Vec<u64>,
    pos: usize,
    capacity: usize,
}

impl BloomFilter {
    fn new(capacity: usize) -> Self {
        let words = (BLOOM_SIZE + 63) / 64;
        Self { bits: vec![0; words], ring: vec![0; capacity], pos: 0, capacity }
    }

    fn hash_pair(h: u64) -> (usize, usize) {
        let h1 = h.wrapping_mul(0x9e3779b97f4a7c15);
        let h2 = h.wrapping_mul(0xbf58476d1ce4e5b9);
        (h1 as usize % BLOOM_SIZE, h2 as usize % BLOOM_SIZE)
    }

    fn insert(&mut self, hash: u64) {
        let (i1, i2) = Self::hash_pair(hash);
        self.bits[i1 / 64] |= 1 << (i1 % 64);
        self.bits[i2 / 64] |= 1 << (i2 % 64);
        let old = self.ring[self.pos];
        self.ring[self.pos] = hash;
        self.pos = (self.pos + 1) % self.capacity;
    }

    fn exact_duplicate(&self, hash: u64) -> bool {
        self.ring.contains(&hash)
    }
}

// ------------------------------
// Inverted index (8 bytes, 256 buckets each)
// ------------------------------
#[derive(Serialize, Deserialize, Clone)]
struct InvertedIndex {
    buckets: [Vec<Vec<usize>>; 8],
}

impl InvertedIndex {
    fn new() -> Self {
        const EMPTY: Vec<Vec<usize>> = Vec::new();
        let mut buckets = [EMPTY; 8];
        for i in 0..8 {
            buckets[i] = vec![Vec::new(); 256];
        }
        Self { buckets }
    }

    fn clear(&mut self) {
        for pos in 0..8 {
            for byte in 0..256 {
                self.buckets[pos][byte].clear();
            }
        }
    }

    fn insert(&mut self, hash: u64, idx: usize) {
        let bytes = hash.to_le_bytes();
        for (pos, &byte) in bytes.iter().enumerate() {
            self.buckets[pos][byte as usize].push(idx);
        }
    }

    fn build(&mut self, memories: &[Memory]) {
        self.clear();
        for (idx, mem) in memories.iter().enumerate() {
            self.insert(mem.hash, idx);
        }
    }

    fn query(&self, query_hash: u64, min_common_bytes: usize) -> Vec<usize> {
        let bytes = query_hash.to_le_bytes();
        let mut candidates: Vec<usize> = Vec::new();
        let mut seen = HashSet::new();

        let mut pos_sizes: Vec<_> = (0..8)
            .map(|pos| (pos, self.buckets[pos][bytes[pos] as usize].len()))
            .collect();
        pos_sizes.sort_by_key(|&(_, sz)| sz);

        for (pos, _) in pos_sizes.iter().take(3) {
            for &idx in &self.buckets[*pos][bytes[*pos] as usize] {
                if seen.insert(idx) {
                    candidates.push(idx);
                }
            }
        }
        candidates
    }
}

#[derive(Serialize, Deserialize, Clone)]
struct Memory {
    id: usize,
    text: String,
    hash: u64,
    timestamp: f64,

}

#[derive(Serialize, Deserialize, Clone)]
struct AntiMemory {
    incorrect: String,
    correct: String,
    successes: u32,
    failures: u32,
    created: f64,
    strength: f32,
}

#[derive(Serialize, Deserialize, Clone)]
struct Settings {

    harmony_mode: bool,

impl Default for Settings {

// ------------------------------
// Global state manager (signals-based)
// ------------------------------
#[derive(Clone)]
struct AppState {
    memories: Signal<Vec<Memory>>,
    anti_memories: Signal<Vec<AntiMemory>>,
    personality: Signal<u8>,
    settings: Signal<Settings>,
    bloom: Signal<BloomFilter>,
    inverted_index: Signal<InvertedIndex>,
    importance_heap: Signal<BinaryHeap<Reverse<(u16, usize)>>>,
    cache: Signal<LruCache<u64, Vec<usize>>>,
    next_id: Signal<usize>,
    chat_messages: Signal<Vec<ChatMessage>>,
}

impl AppState {
    fn new() -> Self {
        let memories = use_persistent("ai_companion_memories", || Vec::new());
        let anti_memories = use_persistent("ai_companion_anti", || Vec::new());
        let personality = use_persistent("ai_companion_personality", || 7u8);
        let settings = use_persistent("ai_companion_settings", Settings::default);
        let next_id = use_persistent("ai_companion_next_id", || 0usize);
        let chat_messages = use_signal(Vec::new);

        let bloom = use_signal(|| {
            let mut bf = BloomFilter::new(128);
            for mem in memories.read().iter() {
                bf.insert(mem.hash);
            }
            bf
        });

        let inverted_index = use_signal(|| {
            let mut idx = InvertedIndex::new();
            idx.build(&memories.read());
            idx
        });

        let importance_heap = use_signal(|| {
            let mut heap = BinaryHeap::new();
            for (idx, mem) in memories.read().iter().enumerate() {
                if !mem.permanent {
                    heap.push(Reverse((mem.importance, idx)));
                }
            }
            heap
        });

        let cache = use_signal(|| LruCache::new(NonZeroUsize::new(CACHE_SIZE).unwrap()));

        Self {
            memories, anti_memories, personality, settings, bloom, inverted_index,
            importance_heap, cache, next_id, chat_messages,
        }
    }

    fn now_secs() -> f64 {
        Utc::now().timestamp() as f64
    }

    fn add_memory(&mut self, text: String, importance_f32: f32, permanent: bool) {
        let importance = (importance_f32 * 65535.0) as u16;
        let hash = text_hash(&text);

        // Dedup check
        if self.bloom.read().exact_duplicate(hash) {
            let mut memories = self.memories.write();
            if let Some(pos) = memories.iter().position(|m| m.hash == hash && m.text == text) {
                let mem = &mut memories[pos];
                if importance > mem.importance {
                    mem.importance = importance;
                    mem.timestamp = Self::now_secs();
                    mem.permanent = permanent;
                }
                return;
            }
        }

        let id = *self.next_id.read();
        *self.next_id.write() += 1;

        let mem = Memory {
            id, text: text.clone(), hash, timestamp: Self::now_secs(), importance, permanent,
        };

        // Update structures
        self.bloom.write().insert(hash);
        self.memories.write().push(mem.clone());
        self.inverted_index.write().insert(hash, self.memories.read().len() - 1);

        if !permanent {
            self.importance_heap.write().push(Reverse((importance, self.memories.read().len() - 1)));
        }

        // Prune if needed
        if self.memories.read().len() > MAX_MEM {
            self.prune_low_importance();
        }
    }

    fn prune_low_importance(&mut self) {
        while self.memories.read().len() > MAX_MEM {
            if let Some(Reverse((_, idx))) = self.importance_heap.write().pop() {
                if idx < self.memories.read().len() && !self.memories.read()[idx].permanent {
                    self.memories.write().swap_remove(idx);
                }
            } else {
                break;
            }
        }
        // Rebuild heap and index
        let memories = self.memories.read();
        let mut heap = BinaryHeap::new();
        for (idx, mem) in memories.iter().enumerate() {
            if !mem.permanent {
                heap.push(Reverse((mem.importance, idx)));
            }
        }
        *self.importance_heap.write() = heap;
        self.inverted_index.write().build(&memories);
    }

    fn harmony_score(&self, mem: &Memory, sim: f32, current_time: f64) -> f32 {
        if !self.settings.read().harmony_mode {
            return sim;
        }
        let days = (current_time - mem.timestamp) / 86400.0;
        let recency = (-RECENCY_LAMBDA * days).exp() as f32;
        let importance = mem.importance as f32 / 65535.0;
        0.5 * sim + 0.3 * importance + 0.2 * recency
    }

    fn retrieve(&mut self, query: &str, top_k: usize) -> Vec<String> {
        let current_time = Self::now_secs();
        let query_hash = text_hash(query);

        // Check cache
        if let Some(cached) = self.cache.read().get(&query_hash) {
            let memories = self.memories.read();
            return cached.iter()
                .filter_map(|&id| memories.iter().find(|m| m.id == id))
                .take(top_k)
                .map(|m| m.text.clone())
                .collect();
        }

        let index = self.inverted_index.read();
        let candidates = index.query(query_hash, 2);
        let memories = self.memories.read();

        let mut scored: Vec<(f32, usize)> = candidates.into_iter()
            .map(|idx| {
                let mem = &memories[idx];
                let sim = similarity(query_hash, mem.hash);
                let score = self.harmony_score(mem, sim, current_time);
                (score, idx)
            })
            .collect();

        scored.sort_by(|a, b| b.0.partial_cmp(&a.0).unwrap());

        // Serendipity
        if self.settings.read().harmony_mode && rand::thread_rng().gen_bool(SERENDIPITY_CHANCE) {
            let low_imp: Vec<_> = memories.iter().enumerate()
                .filter(|(_, m)| m.importance < PRUNE_THRESHOLD && !m.permanent)
                .collect();
            if let Some(&(rand_idx, _)) = low_imp.get(rand::thread_rng().gen_range(0..low_imp.len())) {
                scored.insert(1, (0.8, rand_idx));
            }
        }

        let mut results = Vec::new();
        let mut ids = Vec::new();
        for (score, idx) in scored {
            if results.len() >= top_k && score > (1.0 - EARLY_STOP_THRESHOLD) {
                break;
            }
            results.push(memories[idx].text.clone());
            ids.push(memories[idx].id);
        }

        self.cache.write().put(query_hash, ids);
        results
    }

    fn add_anti_memory(&mut self, incorrect: String, correct: String) {
        self.anti_memories.write().push(AntiMemory {
            incorrect, correct, successes: 1, failures: 1,
            created: Self::now_secs(), strength: 1.0,
        });
    }

    fn get_warnings(&self) -> Vec<String> {
        self.anti_memories.read().iter()
            .filter(|a| a.strength > 0.5)
            .map(|a| format!("Do NOT say '{}'. Instead say '{}'.", a.incorrect, a.correct))
            .collect()
    }
}

#[component]
fn SidebarButton(route: Route, icon: String, label: String, active: bool) -> Element {
    let nav = use_navigator();
    let current_route = use_route::<Route>();

    let is_active = match (&current_route, &route) {
        (Route::Chat {}, Route::Chat {}) => true,
        (Route::Settings {}, Route::Settings {}) => true,
        (Route::Profile {}, Route::Profile {}) => true,
        (Route::Backup {}, Route::Backup {}) => true,
        _ => false,
    };

    let class = if is_active {
        "bg-blue-600 text-white w-full text-left px-4 py-3 rounded-2xl cursor-pointer transition hover:bg-blue-700"
    } else {
        "bg-gray-700 text-gray-300 w-full text-left px-4 py-3 rounded-2xl cursor-pointer transition hover:bg-gray-600"
    };

    rsx! {
        button {
            class: "{class}",
            onclick: move |_| nav.push(route.clone()),
            "{icon} {label}"
        }
    }
}

#[component]
fn Chat() -> Element {
    let mut state = use_context::<AppState>();
    let mut input_text = use_signal(String::new);
    let mut is_loading = use_signal(|| false);

    let send_message = move |_| {
        let text = input_text.read().clone();
        if text.trim().is_empty() || *is_loading.read() {
            return;
        }

        *is_loading.write() = true;
        let mut messages = state.chat_messages.write();
        messages.push(ChatMessage { role: "user".to_string(), content: text.clone() });
        *input_text.write() = String::new();

        // Store as memory
        let memory_text = format!("User: {}\nAI: Echo: {}", text, text);
        state.add_memory(memory_text, 0.7, false);

        spawn(async move {
            tokio::time::sleep(std::time::Duration::from_millis(500)).await;
            // Simulate response (integrate LLM here)
            let warnings = state.get_warnings();
            let mut response = format!("Echo: {}", text);
            if !warnings.is_empty() {
                response.push_str("\n\n[Reminders: ");
                response.push_str(&warnings.join("; "));
                response.push(']');
            }
            state.chat_messages.write().push(ChatMessage { role: "ai".to_string(), content: response });
            *is_loading.write() = false;
        });
    };

    let messages = state.chat_messages.read();

    rsx! {
        div { class: "flex flex-col h-full max-w-3xl mx-auto",
            // Chat log
            div { class: "flex-1 overflow-y-auto mb-4 space-y-3",
                for msg in messages.iter() {
                    div {
                        class: if msg.role == "user" {
                            "bg-blue-600 self-end max-w-[80%] p-3 rounded-2xl ml-auto"
                        } else {
                            "bg-gray-700 self-start max-w-[80%] p-3 rounded-2xl"
                        },
                        strong { "{if msg.role == \"user\" { \"You\" } else { \"AI\" }}: " }
                        "{msg.content}"
                    }
                }
                if *is_loading.read() {
                    div { class: "bg-gray-700 self-start max-w-[80%] p-3 rounded-2xl",
                        strong { "AI: " }
                        span { class: "inline-block w-5 h-5 border-2 border-gray-400 border-t-white rounded-full animate-spin ml-2" }
                    }
                }
            }
            // Input area
            div { class: "flex gap-3",
                input {
                    class: "flex-1 px-4 py-3 bg-gray-700 border-none rounded-full text-white focus:ring-2 focus:ring-blue-500 outline-none",
                    placeholder: "Type a message...",
                    value: "{input_text}",
                    oninput: move |e| input_text.set(e.value()),
                    onkeydown: move |e| if e.key() == Key::Enter { send_message(()) },
                }
                button {
                    class: "px-6 py-3 bg-blue-600 rounded-full font-bold hover:bg-blue-700 transition disabled:opacity-50",
                    onclick: send_message,
                    disabled: *is_loading.read(),
                    "Send"
                }
            }
        }
    }
}

#[component]
fn Settings() -> Element {
    let mut state = use_context::<AppState>();
    let personality = state.personality;
    let settings = state.settings;
    let mut anti_incorrect = use_signal(String::new);
    let mut anti_correct = use_signal(String::new);

    let adjust_personality = move |delta: i8| {
        let new = (*personality.read() as i16 + delta as i16).clamp(0, 15);
        *personality.write() = new as u8;
    };

    let add_anti = move |_| {
        let inc = anti_incorrect.read().clone();
        let cor = anti_correct.read().clone();
        if !inc.is_empty() && !cor.is_empty() {
            state.add_anti_memory(inc, cor);
            *anti_incorrect.write() = String::new();
            *anti_correct.write() = String::new();
        }
    };

    let update_temp = move |e: Event<FormData>| {
        let val: f32 = e.value().parse().unwrap_or(0.0);
        settings.write().temperature = val.clamp(0.0, 0.5);
    };

    let toggle_harmony = move |_| {
        settings.write().harmony_mode = !settings.read().harmony_mode;
    };

    rsx! {
        div { class: "max-w-lg mx-auto space-y-6",
            h2 { class: "text-2xl font-bold" "Settings" }

            // Personality
            div { class: "space-y-2",
                h3 { class: "text-lg" "Personality (Hex: {format!(\"{:X}\", personality.read())})" }
                div { class: "flex gap-2",
                    button { class: "px-4 py-2 bg-blue-600 rounded", onclick: move |_| adjust_personality(1), "+ Intensity" }
                    button { class: "px-4 py-2 bg-blue-600 rounded", onclick: move |_| adjust_personality(-1), "- Intensity" }
                }
            }

            hr { class: "border-gray-700" }

            // Harmony Controls
            div { class: "space-y-3",
                h3 { class: "text-lg" "Harmony Controls" }
                div { class: "flex items-center gap-3",
                    span { "Temperature ({settings.read().temperature:.2})" }
                    input {
                        r#type: "range",
                        min: "0", max: "0.5", step: "0.01",
                        value: "{settings.read().temperature}",
                        oninput: update_temp,
                        class: "flex-1"
                    }
                }
                div { class: "flex items-center gap-2",
                    input {
                        r#type: "checkbox",
                        checked: settings.read().harmony_mode,
                        onchange: toggle_harmony,
                    }
                    label { "Harmony Mode (balanced scoring)" }
                    if settings.read().harmony_mode {
                        span { class: "ml-2 px-2 py-1 bg-blue-600/20 rounded-full text-sm", "✨ serendipity" }
                    }
                }
            }

            hr { class: "border-gray-700" }

            // Anti-Memory
            div { class: "space-y-3",
                h3 { class: "text-lg" "Anti‑Memory" }
                input {
                    class: "w-full px-4 py-2 bg-gray-700 rounded-full text-white",
                    placeholder: "Incorrect phrase",
                    value: "{anti_incorrect}",
                    oninput: move |e| anti_incorrect.set(e.value()),
                }
                input {
                    class: "w-full px-4 py-2 bg-gray-700 rounded-full text-white",
                    placeholder: "Correct phrase",
                    value: "{anti_correct}",
                    oninput: move |e| anti_correct.set(e.value()),
                }
                button { class: "px-4 py-2 bg-blue-600 rounded-full", onclick: add_anti, "Add Correction" }
            }
        }
    }
}

| **Cross‑platform** | Single codebase compiles to Windows, Linux, macOS via `dx serve --platform desktop` | No separate frontend/backend builds. |
| **State Management** | `use_signal`, `use_context_provider` | Reactive, fine‑grained updates. |
| **Persistence** | `dioxus_sdk::storage::use_persistent` | Automatic save/load across sessions. |
| **Routing** | `dioxus_router` | Declarative, type‑safe navigation. |
| **Async Tasks** | `spawn` + Tokio runtime | Non‑blocking background work. |
| **UI Styling** | Inline Tailwind‑like classes | No extra CSS build step (optional). |
| **Memory System** | Full‑featured vector store (hash‑based, inverted index, Bloom filter, heap pruning) | Same as original, now integrated in Rust. |
| **Binary Size** | ~5–8 MB | Significantly smaller than Electron. |

This rewrite condenses the Tauri (Rust backend) + BlokJS (frontend) architecture into a single Rust codebase, reducing complexity while preserving every feature of the original AI Companion.
